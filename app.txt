const API_BASE = '/api';

function resolveImageUrl(url) {
  if (!url) return '';
  if (url.startsWith('http://') || url.startsWith('https://') || url.startsWith('data:')) {
    return url;
  }
  if (API_BASE.startsWith('http')) {
    const backendBase = API_BASE.replace('/api', '');
    return backendBase + (url.startsWith('/') ? url : '/' + url);
  }
  return url;
}

window.togglePasswordVisibility = (inputId, iconElement) => {
  const input = document.getElementById(inputId);
  if (!input) return;
  if (input.type === 'password') {
    input.type = 'text';
    iconElement.classList.remove('fa-eye');
    iconElement.classList.add('fa-eye-slash');
  } else {
    input.type = 'password';
    iconElement.classList.remove('fa-eye-slash');
    iconElement.classList.add('fa-eye');
  }
};

function formatFractions(text) {
  if (!text) return text;
  // Format standalone fractions as vertical fractions
  return text.replace(/\b(\d+)\/(\d+)\b/g, '<span class="cbt-frac"><span>$1</span><span>$2</span></span>');
}

function getMapelSingkatan(name) {
  if (!name) return '-';
  const map = {
    'Pendidikan Agama Islam dan Budi Pekerti': 'PAI & BP',
    'Pendidikan Agama Kristen dan Budi Pekerti': 'PAK & BP',
    'Pendidikan Agama Katolik dan Budi Pekerti': 'PAK & BP',
    'Pendidikan Agama Hindu dan Budi Pekerti': 'PAH & BP',
    'Pendidikan Agama Buddha dan Budi Pekerti': 'PAB & BP',
    'Pendidikan Agama Khonghucu dan Budi Pekerti': 'PAK & BP',
    'Pendidikan Pancasila dan Kewarganegaraan': 'PPKn',
    'Pendidikan Jasmani Olahraga dan Kesehatan': 'PJOK',
    'Seni Budaya dan Prakarya': 'SBdP',
    'Ilmu Pengetahuan Alam': 'IPA',
    'Ilmu Pengetahuan Sosial': 'IPS',
    'Bahasa Indonesia': 'B. Indo',
    'Bahasa Inggris': 'B. Inggris',
    'Matematika': 'MTK'
  };
  for (let key in map) {
    if (name.toLowerCase().includes(key.toLowerCase()) || key.toLowerCase().includes(name.toLowerCase())) return map[key];
  }
  if (name.length > 15) {
     return name.split(' ').filter(w => w.length > 2 && w.toLowerCase() !== 'dan').map(w => w[0].toUpperCase()).join('');
  }
  return name;
}

async function loadSettings() {

function applyGlobalSettings(settings) {
  state.settings = settings;
  
  // 1. Update Document Title
  document.title = `${settings.cbtTitle} | ${settings.schoolName}`;
  
  // 2. Update h1 and h2 headers
  const loginBrandH1 = document.querySelector('.login-brand h1');
  if (loginBrandH1) loginBrandH1.textContent = settings.cbtTitle;
  
  const loginBrandP = document.querySelector('.login-brand p');
  if (loginBrandP) loginBrandP.textContent = settings.schoolName;
  
  const brandHeaders = document.querySelectorAll('.header-brand h2');
  brandHeaders.forEach(el => {
    el.textContent = settings.cbtTitle;
  });
  
  // 3. Update Logos
  const logoElements = document.querySelectorAll('.brand-logo img, .header-logo');
  logoElements.forEach(el => {
    el.src = resolveImageUrl(settings.schoolLogo);
  });
  
  // Update Favicon
  const favicon = document.querySelector('link[rel="icon"]');
  if (favicon) {
    favicon.href = resolveImageUrl(settings.schoolLogo);
  }
}

function initTheme() {
  if (state.activeTheme === 'dark') {
    document.body.classList.add('dark-theme');
    document.body.classList.remove('light-theme');
  } else {
    document.body.classList.add('light-theme');
    document.body.classList.remove('dark-theme');
  }
  updateThemeToggleButtons();
}

function toggleTheme() {
  state.activeTheme = state.activeTheme === 'light' ? 'dark' : 'light';
  localStorage.setItem('cbt_theme', state.activeTheme);
  initTheme();
}

function updateThemeToggleButtons() {
  const btnLog = document.getElementById('btn-toggle-theme-login');
  const navLogs = document.querySelectorAll('.nav-theme-toggle');

  if (state.activeTheme === 'dark') {
    if (btnLog) btnLog.innerHTML = '<i class="fa-solid fa-sun"></i> Mode Terang';
    navLogs.forEach(btn => btn.innerHTML = '<i class="fa-solid fa-sun"></i>');
  } else {
    if (btnLog) btnLog.innerHTML = '<i class="fa-solid fa-moon"></i> Mode Gelap';
    navLogs.forEach(btn => btn.innerHTML = '<i class="fa-solid fa-moon"></i>');
  }
}

function showToast(message, type = 'success') {
  const container = document.getElementById('toast-container');
  if (!container) return;

  const toast = document.createElement('div');
  toast.className = `toast toast-${type}`;
  
  let icon = 'fa-circle-info';
  if (type === 'success') icon = 'fa-circle-check';
  if (type === 'error') icon = 'fa-circle-xmark';
  if (type === 'warning') icon = 'fa-triangle-exclamation';

  toast.innerHTML = `<i class="fa-solid ${icon}"></i> <span>${message}</span>`;
  container.appendChild(toast);

  // Auto remove
  setTimeout(() => {
    toast.style.animation = 'slideIn 0.3s reverse forwards';
    setTimeout(() => toast.remove(), 300);
  }, 4000);
}

// API CALL HELPER WITH AUTH HEADERS
// ----------------------------------------
async function apiCall(endpoint, method = 'GET', body = null) {
  const headers = {
    'Content-Type': 'application/json'
  };
  if (state.token) {
    headers['Authorization'] = state.token;
  }

  const config = { method, headers };
  if (body) {
    config.body = JSON.stringify(body);
  }

  try {
    const res = await fetch(`${API_BASE}${endpoint}`, config);
    if (!res.ok) {
      if (res.status === 401) {
        clearSession();
        loadDataAndRoute();
        showToast('Sesi Anda telah berakhir atau akun Anda telah login di perangkat lain.', 'danger');
        throw new Error('Sesi Anda telah berakhir.');
      }
      const err = await res.json().catch(() => ({ message: 'Terjadi kesalahan server.' }));
      throw new Error(err.message || 'Koneksi gagal.');
    }
    // Handle file downloads
    const contentType = res.headers.get('content-type');
    if (contentType && (contentType.includes('text/csv') || contentType.includes('spreadsheetml'))) {
      return res;
    }
    return await res.json();
  } catch (error) {
    if (error.message !== 'Sesi Anda telah berakhir.') {
      showToast(error.message, 'danger');
    }
    throw error;
  }
}

// Register a custom BlockEmbed for rendering pasted HTML tables in Quill
if (typeof Quill !== 'undefined') {
  const BlockEmbed = Quill.import('blots/block/embed');

  class TableHtmlBlot extends BlockEmbed {
    static create(value) {
      const node = super.create();
      node.innerHTML = value;
      
      // Auto-apply clean styles for preview
      const tables = node.querySelectorAll('table');
      tables.forEach(t => {
        t.setAttribute('border', '1');
        t.style.borderCollapse = 'collapse';
        t.style.width = '100%';
        t.style.marginTop = '8px';
        t.style.marginBottom = '8px';
        t.querySelectorAll('td, th').forEach(c => {
          c.style.border = '1px solid var(--border-color, #cbd5e0)';
          c.style.padding = '6px 10px';
        });
      });
      return node;
    }

    static value(node) {
      return node.innerHTML;
    }
  }
  TableHtmlBlot.blotName = 'table-html';
  TableHtmlBlot.tagName = 'div';
  TableHtmlBlot.className = 'quill-table-html-wrapper';

  Quill.register(TableHtmlBlot);

  // Register the ImageResize module if it exists globally
  const imgResizeModule = window.ImageResize?.default || window.ImageResize;
  if (imgResizeModule) {
    Quill.register('modules/imageResize', imgResizeModule);
  }
}

async function loadDataAndRoute() {
  // Load global settings first
  await loadSettings();

  // Hide all screens
  document.querySelectorAll('.view').forEach(view => view.classList.add('hidden'));

  if (!state.token) {
    // Show login screen
    document.getElementById('login-view').classList.remove('hidden');
    loadLoginClasses();
    loadPublicTeachers();
    return;
  }

  if (state.role === 'admin') {
    document.getElementById('admin-view').classList.remove('hidden');
    document.getElementById('admin-display-name').textContent = toTitleCase(state.user.name || 'Admin');
    loadAllClassesDropdowns();
    const lastAdminTab = localStorage.getItem('cbt_last_tab_admin') || 'admin-dashboard';
    showTab(lastAdminTab);
    if (lastAdminTab === 'admin-dashboard') loadAdminStats();
  } else if (state.role === 'guru') {
    document.getElementById('teacher-view').classList.remove('hidden');
    
    // Sync navbar with latest profile automatically
    apiCall('/teacher/profile').then(profile => {
      if (profile && profile.name) {
        state.user.name = profile.name;
        state.user.type = profile.type;
        localStorage.setItem('cbt_user', JSON.stringify(state.user));
        const typeLabel = profile.type === 'kelas' ? 'Guru Kelas' : 'Guru Mapel';
        document.getElementById('teacher-display-name').innerHTML = `<div style="line-height: 1.2; text-align: right;"><div>${toTitleCase(profile.name || 'Guru')}</div><div style="color: #3b82f6; font-size: 0.85em; margin-top: 2px;">${typeLabel}</div></div>`;
      }
    }).catch(() => {
      const typeLabel = state.user.type === 'kelas' ? 'Guru Kelas' : 'Guru Mapel';
      document.getElementById('teacher-display-name').innerHTML = `<div style="line-height: 1.2; text-align: right;"><div>${toTitleCase(state.user.name || 'Guru')}</div><div style="color: #3b82f6; font-size: 0.85em; margin-top: 2px;">${typeLabel}</div></div>`;
    });
    
    // Hide Student management sidebar option for subject teachers (guru mapel)
    const studentMenuBtn = document.querySelector('#teacher-view .sidebar-nav button[data-tab="teacher-students"]');
    if (studentMenuBtn) {
      if (state.user.type === 'mapel') {
        studentMenuBtn.classList.add('hidden');
      } else {
        studentMenuBtn.classList.remove('hidden');
      }
    }
    
    loadAllClassesDropdowns();
    let lastTeacherTab = localStorage.getItem('cbt_last_tab_guru') || 'teacher-dashboard';
    if (state.user.type === 'mapel' && lastTeacherTab === 'teacher-students') {
      lastTeacherTab = 'teacher-dashboard';
    }
    showTab(lastTeacherTab);
  } else if (state.role === 'murid') {
    document.getElementById('student-view').classList.remove('hidden');
    document.getElementById('student-display-name-exam').textContent = toTitleCase(state.user.name);
    document.getElementById('student-display-class-exam').textContent = state.user.classId;
    
    // Populate the new minimal identity card fields
    if (document.getElementById('token-student-name')) {
      document.getElementById('token-student-name').textContent = state.user.name;
    }
    if (document.getElementById('token-student-class')) {
      document.getElementById('token-student-class').textContent = state.user.classId;
    }
    if (document.getElementById('token-student-nis')) {
      document.getElementById('token-student-nis').textContent = state.user.nis;
    }
    if (document.getElementById('token-student-nisn')) {
      document.getElementById('token-student-nisn').textContent = state.user.nisn || '-';
    }
    
    // Set instructions text
    if (state.settings && document.getElementById('exam-instructions-text')) {
      document.getElementById('exam-instructions-text').textContent = state.settings.examInstructions;
    }
    
    // Check if there is an active exam already stored in session
    const storedExam = localStorage.getItem('cbt_active_exam_data');
    const storedAttemptId = localStorage.getItem('cbt_active_attempt_id');
    
    if (storedExam && storedAttemptId) {
      state.currentExam = JSON.parse(storedExam);
      state.activeAttemptId = storedAttemptId;
      startExamLogic(state.currentExam.examId, true); // Resume exam
    } else {
      // Normal token verification state
      const examHeader = document.querySelector('.exam-header');
      if (examHeader) examHeader.classList.add('hidden');

      document.getElementById('exam-token-verification').classList.remove('hidden');
      document.getElementById('exam-main-interface').classList.add('hidden');

      // Clear input fields and hide selected exam info
      if (document.getElementById('exam-token-input')) {
        document.getElementById('exam-token-input').value = '';
      }
      if (document.getElementById('selected-exam-info')) {
        document.getElementById('selected-exam-info').classList.add('hidden');
      }
    }
  }
}

// TAB MANAGEMENT
// ----------------------------------------
function showTab(tabId) {
  const container = document.getElementById(tabId).closest('.app-container');
  
  // Deactivate all tab panes in this container
  container.querySelectorAll('.tab-pane').forEach(pane => pane.classList.remove('active'));
  // Deactivate all sidebar nav buttons in this container
  container.querySelectorAll('.nav-item').forEach(btn => btn.classList.remove('active'));

  // Activate selected
  const activePane = document.getElementById(tabId);
  if (activePane) activePane.classList.add('active');
  
  const navBtn = container.querySelector(`.nav-item[data-tab="${tabId}"]`);
  if (navBtn) navBtn.classList.add('active');

  if (state.role) {
    localStorage.setItem(`cbt_last_tab_${state.role}`, tabId);
  }

  // Trigger tab-specific loads
  if (tabId === 'admin-dashboard') {
    stopAdminMonitoring();
    loadAdminStats();
  }
  if (tabId === 'admin-teachers') {
    stopAdminMonitoring();
    loadAdminTeachers();
  }
  if (tabId === 'admin-classes') {
    stopAdminMonitoring();
    loadAdminClasses();
    loadAdminMappings();
  }
  if (tabId === 'admin-subjects') {
    stopAdminMonitoring();
    loadAdminSubjects();
  }
  if (tabId === 'admin-reports') {
    startAdminMonitoring();
  }
  if (tabId === 'admin-settings') {
    stopAdminMonitoring();
    loadAdminSettingsForm();
  }

  if (tabId === 'teacher-dashboard') {
    stopMonitoringLive();
    startTeacherDashboardPolling();
  }
  if (tabId === 'teacher-students') {
    stopMonitoringLive();
    stopTeacherDashboardPolling();
    loadTeacherStudents();
  }
  if (tabId === 'teacher-questions') {
    stopMonitoringLive();
    stopTeacherDashboardPolling();
    loadTeacherQuestions();
  }
  if (tabId === 'teacher-exams') {
    stopMonitoringLive();
    stopTeacherDashboardPolling();
    loadTeacherExams();
  }
  if (tabId === 'teacher-monitoring') {
    stopTeacherDashboardPolling();
    startMonitoringLive();
  }
  if (tabId === 'teacher-grading') {
    stopMonitoringLive();
    stopTeacherDashboardPolling();
    loadTeacherAttempts();
  }
  if (tabId === 'teacher-profile') {
    stopMonitoringLive();
    stopTeacherDashboardPolling();
    loadTeacherProfile();
  }
}

// USER INTERFACE EVENT LISTENERS
// ----------------------------------------
function setupEventListeners() {
  // Theme Toggles
  const btnToggleThemeLogin = document.getElementById('btn-toggle-theme-login');
  if (btnToggleThemeLogin) {
    btnToggleThemeLogin.addEventListener('click', toggleTheme);
  }

  document.querySelectorAll('.nav-theme-toggle').forEach(btn => {
    btn.addEventListener('click', toggleTheme);
  });

  // Login Role Tabs switcher
  const loginTabs = document.querySelectorAll('.login-tab');
  loginTabs.forEach(tab => {
    tab.addEventListener('click', () => {
      loginTabs.forEach(t => t.classList.remove('active'));
      tab.classList.add('active');

      const role = tab.getAttribute('data-role');
      document.querySelectorAll('.login-form').forEach(f => f.classList.add('hidden'));

      if (role === 'murid') document.getElementById('form-login-murid').classList.remove('hidden');
      if (role === 'guru') document.getElementById('form-login-guru').classList.remove('hidden');
      if (role === 'admin') document.getElementById('form-login-admin').classList.remove('hidden');
    });
  });

  // Anti Cheat toggle in Exam Modal
  const chkAntiCheat = document.getElementById('exam-anti-cheat-enabled');
  if (chkAntiCheat) {
    chkAntiCheat.addEventListener('change', (e) => {
      const container = document.getElementById('exam-unlock-key-container');
      const inputKey = document.getElementById('exam-unlock-key');
      if (e.target.checked) {
        container.style.display = 'block';
        if (inputKey) inputKey.required = true;
      } else {
        container.style.display = 'none';
        if (inputKey) inputKey.required = false;
      }
    });
  }

  // Student login dynamic names based on class
  document.getElementById('murid-class').addEventListener('change', async (e) => {
    const classId = e.target.value;
    const nameSelect = document.getElementById('murid-name');
    nameSelect.innerHTML = '<option value="" disabled selected>Memuat nama siswa...</option>';
    try {
      const res = await fetch(`${API_BASE}/public/students?classId=${encodeURIComponent(classId)}`);
      if (!res.ok) throw new Error();
      const students = await res.json();
      nameSelect.innerHTML = '<option value="" disabled selected>Pilih nama Anda</option>';
      students.forEach(s => {
        nameSelect.innerHTML += `<option value="${s.name}">${s.name}</option>`;
      });
      nameSelect.removeAttribute('disabled');
    } catch(err) {
      nameSelect.innerHTML = '<option value="" disabled selected>Gagal memuat nama</option>';
    }
  });

  // Logins Submit Form Handlers
  document.getElementById('form-login-murid').addEventListener('submit', async (e) => {
    e.preventDefault();
    const name = document.getElementById('murid-name').value;
    const classId = document.getElementById('murid-class').value;
    const nisn = document.getElementById('murid-nisn').value;

    try {
      const res = await apiCall('/auth/login', 'POST', { role: 'murid', name, classId, nisn });
      saveSession(res.token, res.role, res.user);
      loadDataAndRoute();
    } catch (err) {}
  });

  document.getElementById('form-login-guru').addEventListener('submit', async (e) => {
    e.preventDefault();
    const username = document.getElementById('guru-username').value;
    const password = document.getElementById('guru-password').value;

    try {
      const res = await apiCall('/auth/login', 'POST', { role: 'guru', username, password });
      saveSession(res.token, res.role, res.user);
      loadDataAndRoute();
    } catch (err) {}
  });

  document.getElementById('form-login-admin').addEventListener('submit', async (e) => {
    e.preventDefault();
    const username = document.getElementById('admin-username').value;
    const password = document.getElementById('admin-password').value;

    try {
      const res = await apiCall('/auth/login', 'POST', { role: 'admin', username, password });
      saveSession(res.token, res.role, res.user);
      loadDataAndRoute();
    } catch (err) {}
  });

  document.querySelectorAll('.btn-logout').forEach(btn => {
    btn.addEventListener('click', () => {
      clearSession();
      loadDataAndRoute();
    });
  });

  // Generic Modal Close buttons
  document.querySelectorAll('.modal-close').forEach(btn => {
    btn.addEventListener('click', (e) => {
      const modalId = btn.getAttribute('data-modal') || btn.closest('.modal').id;
      closeModal(modalId);

  });

    btnStartCam.addEventListener('click', startAttendanceCamera);
  }

              currentIndex += 1;
            });
            questionQuill.setSelection(currentIndex);
            showToast('Tabel berhasil ditempel!', 'success');
            return;
          }
        }
      }

      // 2. Check if clipboard contains image data
      const items = (e.clipboardData || e.originalEvent.clipboardData).items;
      let hasImage = false;
      let imageFile = null;
      for (let i = 0; i < items.length; i++) {
        if (items[i].type.indexOf('image') === 0) {
          hasImage = true;
          imageFile = items[i].getAsFile();
          break;
        }
      }
      
      if (!hasImage || !imageFile) return;

      // Prevent default pasting behavior
      e.preventDefault();

      // Check focus context
      const activeEl = document.activeElement;
      const isQuillFocused = activeEl && (activeEl.closest('#question-text-editor') || activeEl.classList.contains('ql-editor'));

      formData.append('image', imageFile);

      try {
        showToast('Mengunggah gambar yang ditempel...', 'info');
        const token = state.token || localStorage.getItem('cbt_token');
        const response = await fetch(`${API_BASE}/upload-image`, {
          method: 'POST',
          headers: { 'Authorization': token },
          body: formData
        });
        
        if (!response.ok) throw new Error('Gagal mengunggah gambar yang disalin.');
        const resData = await response.json();
        
        if (isQuillFocused && questionQuill) {
          // Insert inline into Quill editor
          const range = questionQuill.getSelection(true);
          const resolvedUrl = resolveImageUrl(resData.imageUrl);
          questionQuill.insertEmbed(range.index, 'image', resolvedUrl);
          questionQuill.setSelection(range.index + 1);
          showToast('Gambar berhasil ditempel ke pertanyaan!', 'success');
        } else if (isOptionValFocused) {
          // Paste to focused option input image field
          const idx = activeEl.getAttribute('data-idx');
          if (idx !== null) {
            const container = document.getElementById('options-dynamic-fields');
            const imgValInput = container.querySelector(`.pg-option-img-val[data-idx="${idx}"]`);
            if (imgValInput) {
              imgValInput.value = resData.imageUrl;
              const previewContainer = container.querySelector(`.option-image-preview-container-${idx}`);
              if (previewContainer) {
                previewContainer.querySelector(`.option-img-preview-${idx}`).src = resolveImageUrl(resData.imageUrl);
                previewContainer.style.display = 'flex';
              }
              showToast(`Gambar berhasil ditempel ke Opsi ${String.fromCharCode(65 + parseInt(idx, 10))}!`, 'success');
            }
          }
        } else {
          // Default: paste as Stimulus Image
          document.getElementById('question-image-url').value = resData.imageUrl;
          updateStimulusPreview(resData.imageUrl);
          showToast('Gambar berhasil ditempel sebagai stimulus!', 'success');
        }
      } catch (err) {
        console.error('Error pasting image:', err);
        showToast(err.message || 'Terjadi kesalahan saat menempel gambar', 'danger');
      }
    });
  }
}

function saveSession(token, role, user) {
  state.token = token;
  state.role = role;
  state.user = user;
  localStorage.setItem('cbt_token', token);
  localStorage.setItem('cbt_role', role);
  localStorage.setItem('cbt_user', JSON.stringify(user));
}

function clearSession() {
  state.token = '';
  state.role = '';
  state.user = null;
  state.currentExam = null;
  state.questions = [];
  state.activeQuestionIndex = 0;
  state.studentAnswers = {};
  state.studentDoubtQuestions = {};
  state.activeAttemptId = null;

  clearInterval(state.examTimerInterval);
  clearInterval(state.monitoringInterval);
  if (state.teacherDashboardInterval) {
    clearInterval(state.teacherDashboardInterval);
    state.teacherDashboardInterval = null;
  }
  if (state.adminMonitoringInterval) {
    clearInterval(state.adminMonitoringInterval);
    state.adminMonitoringInterval = null;
  }

  localStorage.removeItem('cbt_token');
  localStorage.removeItem('cbt_role');
  localStorage.removeItem('cbt_user');
  localStorage.removeItem('cbt_active_exam_data');
  localStorage.removeItem('cbt_active_attempt_id');
}

window.toggleRowPassword = (btn) => {
  const cell = btn.closest('td');
  const txt = cell.querySelector('.pw-text');
  const masked = cell.querySelector('.pw-masked');
  const icon = btn.querySelector('i');
  
  if (txt.style.display === 'none') {
    txt.style.display = 'inline';
    masked.style.display = 'none';
    icon.classList.remove('fa-eye-slash');
    icon.classList.add('fa-eye');
  } else {
    txt.style.display = 'none';
    masked.style.display = 'inline';
    icon.classList.remove('fa-eye');
    icon.classList.add('fa-eye-slash');
  }
};

async function loadAdminMappings() {
  try {
    const list = await apiCall('/admin/mappings');
    const container = document.getElementById('admin-mappings-list');
    if (!container) return;
    container.innerHTML = '';
    
    // Group by teacherName
    const grouped = {};
    list.forEach(m => {
      if (!grouped[m.teacherName]) {
        grouped[m.teacherName] = [];
      }
      grouped[m.teacherName].push(m);
    });
    
    const teacherEntries = Object.entries(grouped);
    const totalTeachers = teacherEntries.length;
    
    if (totalTeachers === 0) {
      container.innerHTML = '<div class="glass" style="text-align: center; padding: 2rem;">Belum ada data pemetaan.</div>';
      const pagContainer = document.getElementById('mappings-pagination-container');
      if (pagContainer) pagContainer.innerHTML = '';
      return;
    }
    
    const totalPages = Math.ceil(totalTeachers / teachersPerPage) || 1;
    if (teachersCurrentPage > totalPages) {
      teachersCurrentPage = totalPages;
    }
    if (teachersCurrentPage < 1) {
      teachersCurrentPage = 1;
    }
    
    const startIndex = (teachersCurrentPage - 1) * teachersPerPage;
    const endIndex = startIndex + teachersPerPage;
    const paginatedTeachers = teacherEntries.slice(startIndex, endIndex);
    
    paginatedTeachers.forEach(([teacherName, mappings]) => {
      let rowsHtml = '';
      mappings.forEach(m => {
        rowsHtml += `
          <div style="display: flex; justify-content: space-between; align-items: center; padding: 0.75rem 0; border-bottom: 1px solid var(--color-border);">
            <div>
              <span class="badge badge-primary" style="margin-right: 0.5rem;">${m.subjectName}</span>
              <span class="badge badge-dark">${m.classId}</span>
            </div>
            <button class="btn btn-sm btn-outline-danger" title="Hapus" onclick="deleteMapping('${m.id}')"><i class="fa-solid fa-trash"></i></button>
          </div>
        `;
      });
      
      container.innerHTML += `
        <details class="glass" style="margin-bottom: 0.5rem; border-radius: var(--radius-md); overflow: hidden;">
          <summary style="padding: 0.6rem 1rem; cursor: pointer; font-weight: 500; font-size: 0.95rem; background: rgba(255, 255, 255, 0.5);">
            <i class="fa-solid fa-chalkboard-user" style="color: var(--color-primary); margin-right: 0.5rem;"></i>
            ${teacherName}
            <span class="badge badge-light" style="font-size: 0.85rem; float: right; margin-top: 2px;">${mappings.length} Pemetaan</span>
          </summary>
          <div style="padding: 0 1rem 0.5rem 1rem; border-top: 1px solid var(--color-border);">
            ${rowsHtml}
          </div>
        </details>
      `;
    });
    
    // Render pagination controls
    const pagContainer = document.getElementById('mappings-pagination-container');
    if (pagContainer) {
      pagContainer.innerHTML = `
        <div class="pagination-controls" style="display: flex; justify-content: space-between; align-items: center; margin-top: 1rem; padding: 0.5rem 0;">
          <button class="btn btn-sm btn-outline-primary" ${teachersCurrentPage === 1 ? 'disabled' : ''} onclick="changeTeachersPage(-1)"><i class="fa-solid fa-chevron-left"></i> Sebelumnya</button>
          <span style="font-size: 0.85rem; color: var(--text-muted); font-weight: 500;">Halaman ${teachersCurrentPage} dari ${totalPages} (Total: ${totalTeachers} Guru)</span>
          <button class="btn btn-sm btn-outline-primary" ${teachersCurrentPage === totalPages ? 'disabled' : ''} onclick="changeTeachersPage(1)">Berikutnya <i class="fa-solid fa-chevron-right"></i></button>
        </div>
      `;
    }
  } catch (err) {}
}

async function loadAdminSubjects() {
  try {
    const list = await apiCall('/admin/subjects');
    const tbody = document.getElementById('table-subjects-body');
    if (!tbody) return;
    tbody.innerHTML = '';
    list.forEach(s => {
      tbody.innerHTML += `
        <tr draggable="true" data-id="${s.id}">
          <td class="drag-handle"><i class="fa-solid fa-grip-vertical text-muted drag-trigger" style="cursor: move;"></i></td>
          <td>${s.id}</td>
          <td>${s.name}</td>
          <td class="text-right" style="white-space: nowrap;">
            <button class="btn btn-sm btn-outline-warning" title="Ubah" onclick="editSubject('${s.id}', '${s.name}')"><i class="fa-solid fa-pen"></i></button>
            <button class="btn btn-sm btn-outline-danger" title="Hapus" onclick="deleteSubject('${s.id}', '${s.name}')"><i class="fa-solid fa-trash"></i></button>
          </td>
        </tr>
      `;
    });

    // Add drag and drop listeners
    const rows = tbody.querySelectorAll('tr');
    rows.forEach(row => {
      row.addEventListener('dragstart', (e) => {
        row.classList.add('dragging');
        e.dataTransfer.effectAllowed = 'move';
      });

      row.addEventListener('dragover', (e) => {
        e.preventDefault();
        const draggingRow = tbody.querySelector('.dragging');
        const targetRow = e.target.closest('tr');
        if (targetRow && targetRow !== draggingRow) {
          const rect = targetRow.getBoundingClientRect();
          const next = (e.clientY - rect.top) / (rect.bottom - rect.top) > 0.5;
          tbody.insertBefore(draggingRow, next ? targetRow.nextSibling : targetRow);
        }
      });

      row.addEventListener('dragend', async () => {
        row.classList.remove('dragging');
        
        // Gather new order
        const reorderedRows = Array.from(tbody.querySelectorAll('tr'));
        const subjectIds = reorderedRows.map(r => r.getAttribute('data-id'));
        
        try {
          await apiCall('/admin/subjects/reorder', 'POST', { subjectIds });
          showToast('Urutan mata pelajaran diperbarui.');
        } catch (err) {
          // Re-load to revert order if error
          loadAdminSubjects();
        }
      });
    });

  } catch (err) {}
}

async function loadAllClassesDropdowns() {
  try {
    let classes = await apiCall('/classes');
    console.log('[DEBUG] loadAllClassesDropdowns: initial classes =', classes, 'state.role =', state.role, 'state.user =', state.user);
    
    // Filter classes to only include mapped ones if the user is a teacher
    if (state.role === 'guru') {
      const mappings = await apiCall('/admin/mappings');
      console.log('[DEBUG] loadAllClassesDropdowns: mappings =', mappings);
      const teacherMappings = mappings.filter(m => m.teacherId === state.user.id);
      console.log('[DEBUG] loadAllClassesDropdowns: teacherMappings =', teacherMappings);
      const mappedClassIds = [...new Set(teacherMappings.map(m => m.classId))];
      console.log('[DEBUG] loadAllClassesDropdowns: mappedClassIds =', mappedClassIds);
      classes = classes.filter(c => mappedClassIds.includes(c));
      console.log('[DEBUG] loadAllClassesDropdowns: filtered classes =', classes);
    }
    
    // 1. Student Login class select
    const loginSelect = document.getElementById('murid-class');
    if (loginSelect) {
      const currentVal = loginSelect.value;
      loginSelect.innerHTML = '<option value="" disabled selected>Pilih kelas Anda</option>';
      classes.forEach(c => {
        loginSelect.innerHTML += `<option value="${c}">${c}</option>`;
      });
      if (currentVal) loginSelect.value = currentVal;
    }

    // 2. Student Add/Edit modal class select
    const studentSelect = document.getElementById('student-class-input');
    if (studentSelect) {
      const currentVal = studentSelect.value;
      studentSelect.innerHTML = '';
      classes.forEach(c => {
        studentSelect.innerHTML += `<option value="${c}">${c}</option>`;
      });
      if (currentVal) studentSelect.value = currentVal;
    }

    // 2.5 Student Import modal class select
    const importSelect = document.getElementById('import-student-class-select');
    if (importSelect) {
      const currentVal = importSelect.value;
      importSelect.innerHTML = '';
      classes.forEach(c => {
        importSelect.innerHTML += `<option value="${c}">${c}</option>`;
      });
      if (currentVal) importSelect.value = currentVal;
      
      const updateTemplateHref = () => {
        const btn = document.getElementById('btn-download-template-student');
        if (btn && importSelect.value) {
          btn.href = `/api/teacher/students/template?classId=${encodeURIComponent(importSelect.value)}`;
        }
      };
      importSelect.removeEventListener('change', updateTemplateHref);
      importSelect.addEventListener('change', updateTemplateHref);
      updateTemplateHref();
    }

    // 2.6 Student table filter
    const filterStudentSelect = document.getElementById('filter-student-class');
    if (filterStudentSelect) {
      const currentVal = filterStudentSelect.value;
      filterStudentSelect.innerHTML = '<option value="ALL">Semua Kelas</option>';
      classes.forEach(c => {
        filterStudentSelect.innerHTML += `<option value="${c}">Kelas ${c}</option>`;
      });
      if (currentVal && (currentVal === 'ALL' || classes.includes(currentVal))) {
        filterStudentSelect.value = currentVal;
      }
      
      filterStudentSelect.removeEventListener('change', loadTeacherStudents);
      filterStudentSelect.addEventListener('change', loadTeacherStudents);
    }

    // 3. Mapping modal class checkboxes
    const classContainer = document.getElementById('mapping-classes-checkboxes');
    if (classContainer) {
      classContainer.innerHTML = '';
      classes.forEach(c => {
        classContainer.innerHTML += `
          <label class="checkbox-item">
            <input type="checkbox" name="mapping-classes" value="${c}">
            <span>${c}</span>
          </label>
        `;
      });
    }

    // 4. Exam Scheduler class select
    const examSelect = document.getElementById('exam-class-select');
    if (examSelect) {
      const currentVal = examSelect.value;
      examSelect.innerHTML = '';
      classes.forEach(c => {
        examSelect.innerHTML += `<option value="${c}">${c}</option>`;
      });
      if (currentVal) examSelect.value = currentVal;
    }
  } catch (err) {
    console.error('Gagal memuat daftar kelas untuk dropdown:', err);
  }
}

window.uploadStudentPhoto = async function(inputElem) {
  if (!inputElem.files || inputElem.files.length === 0) return;
  const file = inputElem.files[0];
  const formData = new FormData();
  formData.append('image', file);
  
  const statusElem = document.getElementById('student-photo-status');
  if (statusElem) statusElem.textContent = 'Mengunggah...';
  
  try {
    const token = state.token || localStorage.getItem('cbt_token');
    const response = await fetch(`${API_BASE}/upload-image`, {
      method: 'POST',
      headers: { 'Authorization': token },
      body: formData
    });
    
    if (response.ok) {
      const data = await response.json();
      document.getElementById('student-photo-input').value = data.imageUrl;
      const preview = document.getElementById('student-photo-preview');
      if (preview) {
        preview.src = resolveImageUrl(data.imageUrl);
        preview.style.display = 'block';
      }
      if (statusElem) {
        statusElem.innerHTML = '<i class="fa-solid fa-check text-success"></i> Berhasil diunggah';
        setTimeout(() => statusElem.textContent = '', 3000);
      }
    } else {
      if (statusElem) statusElem.innerHTML = '<i class="fa-solid fa-xmark text-danger"></i> Gagal mengunggah';
      showToast('Gagal mengunggah gambar', 'danger');
    }
  } catch (error) {
    if (statusElem) statusElem.innerHTML = '<i class="fa-solid fa-xmark text-danger"></i> Terjadi kesalahan';
    console.error('Error uploading student photo:', error);
    showToast('Terjadi kesalahan saat mengunggah', 'danger');
  }
  // Reset the file input so the same file can be selected again if needed
  inputElem.value = '';
};

async function loadTeacherQuestions() {
  try {
    // 1. Fetch teacher mappings
    const mappings = await apiCall('/admin/mappings');
    const teacherMappings = mappings.filter(m => m.teacherId === state.user.id);

    const filterSubjectSelect = document.getElementById('question-filter-subject');
    const filterClassSelect = document.getElementById('question-filter-class');

    // Populate Mata Pelajaran dropdown
    const uniqueSubjects = [];
    const subjectMap = new Map();
    teacherMappings.forEach(m => {
      if (!subjectMap.has(m.subjectId)) {
        subjectMap.set(m.subjectId, m.subjectName);
        uniqueSubjects.push({ id: m.subjectId, name: m.subjectName });
      }
    });

    filterSubjectSelect.innerHTML = '<option value="" disabled selected>Pilih Mata Pelajaran</option>';
    uniqueSubjects.forEach(s => {
      filterSubjectSelect.innerHTML += `<option value="${s.id}">${s.name}</option>`;
    });

    // Clear and disable class dropdown initially
    filterClassSelect.innerHTML = '<option value="" disabled selected>Pilih Kelas</option>';
    filterClassSelect.setAttribute('disabled', 'true');

    // Restore or initialize selections
    state.teacherSelectedPacket = state.teacherSelectedPacket || '';
    state.teacherSelectedSubject = state.teacherSelectedSubject || '';
    state.teacherSelectedClass = state.teacherSelectedClass || '';

    // Restore subject selection
    if (state.teacherSelectedSubject && uniqueSubjects.some(s => s.id === state.teacherSelectedSubject)) {
      filterSubjectSelect.value = state.teacherSelectedSubject;
      
      // Populate Class dropdown based only on mappings for the restored subject
      const subjectClassIds = [...new Set(
        teacherMappings.filter(m => m.subjectId === state.teacherSelectedSubject).map(m => m.classId)
      )];
      filterClassSelect.innerHTML = '<option value="" disabled selected>Pilih Kelas</option>';
      subjectClassIds.forEach(c => {
        filterClassSelect.innerHTML += `<option value="${c}">${c}</option>`;
      });
      filterClassSelect.removeAttribute('disabled');
      
      if (state.teacherSelectedClass && subjectClassIds.includes(state.teacherSelectedClass)) {
        filterClassSelect.value = state.teacherSelectedClass;
      } else {
        state.teacherSelectedClass = '';
      }
      // Load packet options for the restored subject
      await loadPaketDropdownOptions(state.teacherSelectedSubject, state.teacherSelectedPacket);
    } else {
      state.teacherSelectedSubject = '';
      state.teacherSelectedClass = '';
      state.teacherSelectedPacket = '';
      resetPaketDropdown();
    }

    // Bind event listeners (only once)
    if (!filterSubjectSelect.dataset.listenerBound) {
      filterSubjectSelect.dataset.listenerBound = 'true';
      filterSubjectSelect.addEventListener('change', async (e) => {
        const val = e.target.value;
        state.teacherSelectedSubject = val;
        state.teacherSelectedPacket = '';
        state.teacherSelectedClass = '';
        filterClassSelect.value = '';

        if (val) {
          // Re-populate class options for this specific subject mapping
          const subjectClassIds = [...new Set(
            teacherMappings.filter(m => m.subjectId === val).map(m => m.classId)
          )];
          filterClassSelect.innerHTML = '<option value="" disabled selected>Pilih Kelas</option>';
          subjectClassIds.forEach(c => {
            filterClassSelect.innerHTML += `<option value="${c}">${c}</option>`;
          });
          filterClassSelect.removeAttribute('disabled');
          await loadPaketDropdownOptions(val, '');
        } else {
          filterClassSelect.innerHTML = '<option value="" disabled selected>Pilih Kelas</option>';
          filterClassSelect.setAttribute('disabled', 'true');
          resetPaketDropdown();
        }
        updateQuestionBankWorkspaceVisibility();
      });
    }

    if (!filterClassSelect.dataset.listenerBound) {
      filterClassSelect.dataset.listenerBound = 'true';
      filterClassSelect.addEventListener('change', (e) => {
        state.teacherSelectedClass = e.target.value;
        updateQuestionBankWorkspaceVisibility();
      });
    }

    // Initial check/render
    updateQuestionBankWorkspaceVisibility();

  } catch (err) {
    console.error(err);
  }
}

let _cachedPaketList = [];

async function loadPaketDropdownOptions(subjectId, selectedName = '') {
  try {
    const paketList = await apiCall(`/teacher/paket-soal?teacherId=${state.user.id}&subjectId=${subjectId}`);
    _cachedPaketList = paketList;
    renderPaketOptions(paketList, selectedName);
    if (selectedName && paketList.some(p => p.name === selectedName)) {
      selectPaket(selectedName);
    } else {
      state.teacherSelectedPacket = '';
      document.getElementById('question-filter-packet').value = '';
      const label = document.getElementById('paket-selected-label');
      if (label) label.textContent = 'Pilih Paket Soal';
    }
    return paketList;
  } catch(err) {
    console.error(err);
    return [];
  }
}

function renderPaketOptions(paketList, selectedName = '') {
  const list = document.getElementById('paket-options-list');
  if (!list) return;
  list.innerHTML = '';

  if (paketList.length === 0) {
    list.innerHTML = `<div class="paket-empty-msg"><i class="fa-solid fa-folder-open"></i> Belum ada paket soal.<br><small>Klik "Kelola Paket Soal" untuk menambahkan.</small></div>`;
    return;
  }

  paketList.forEach(p => {
    const isActive = p.name === selectedName;
    const item = document.createElement('div');
    item.className = `paket-option-item ${isActive ? 'active' : ''}`;
    item.dataset.name = p.name;
    item.dataset.id = p.id;
    item.innerHTML = `
      <div class="paket-option-info">
        <span class="paket-option-name">${p.name}</span>
        ${p.description ? `<span class="paket-option-desc">${p.description}</span>` : ''}
      </div>
      <i class="fa-solid fa-check paket-check-icon" style="${isActive ? '' : 'display:none;'}"></i>
    `;
    item.addEventListener('click', () => {
      selectPaket(p.name);
      closePaketDropdown();
    });
    list.appendChild(item);
  });
}

function selectPaket(name) {
  state.teacherSelectedPacket = name;
  document.getElementById('question-filter-packet').value = name;
  const label = document.getElementById('paket-selected-label');
  if (label) label.textContent = name || 'Pilih Paket Soal';

  // Update active styles in list
  document.querySelectorAll('.paket-option-item').forEach(item => {
    const isActive = item.dataset.name === name;
    item.classList.toggle('active', isActive);
    const check = item.querySelector('.paket-check-icon');
    if (check) check.style.display = isActive ? '' : 'none';
  });

  updateQuestionBankWorkspaceVisibility();
}

function resetPaketDropdown() {
  state.teacherSelectedPacket = '';
  document.getElementById('question-filter-packet').value = '';
  const label = document.getElementById('paket-selected-label');
  if (label) label.textContent = 'Pilih Paket Soal';
  _cachedPaketList = [];
  const list = document.getElementById('paket-options-list');
  if (list) list.innerHTML = '';
  closePaketDropdown();
}

async function togglePaketDropdown() {
  const subjectId = state.teacherSelectedSubject;
  if (!subjectId) { showToast('Pilih mata pelajaran terlebih dahulu!', 'warning'); return; }
  const menu = document.getElementById('paket-dropdown-menu');
  if (menu) {
    const isOpening = menu.classList.contains('hidden');
    if (isOpening) {
      // Auto fetch latest packets when clicked to open
      await loadPaketDropdownOptions(subjectId, state.teacherSelectedPacket);
    }
    menu.classList.toggle('hidden');
    if (!menu.classList.contains('hidden')) {
      const searchInp = document.getElementById('paket-search-input');
      if (searchInp) { searchInp.value = ''; filterPaketOptions(''); searchInp.focus(); }
    }
  }
}

function closePaketDropdown() {
  const menu = document.getElementById('paket-dropdown-menu');
  if (menu) menu.classList.add('hidden');
}

function filterPaketOptions(query) {
  const lq = query.toLowerCase();
  const items = document.querySelectorAll('.paket-option-item');
  let anyVisible = false;
  items.forEach(item => {
    const name = item.dataset.name.toLowerCase();
    const visible = name.includes(lq);
    item.style.display = visible ? '' : 'none';
    if (visible) anyVisible = true;
  });

  const addRow = document.getElementById('paket-add-row');
  const emptyMsg = document.querySelector('.paket-empty-msg');
  // Show "add" button if query doesn't match any existing and there is content typed
  const exactMatch = _cachedPaketList.some(p => p.name.toLowerCase() === lq);
  if (addRow) {
    addRow.style.display = (query && !exactMatch) ? '' : 'none';
    const btn = document.getElementById('btn-quick-add-paket');
    if (btn && query) btn.innerHTML = `<i class="fa-solid fa-plus"></i> Tambah Paket "${query}"`;
  }
}

const setupSendPacketAutocomplete = () => {
  const input = document.getElementById('send-packet-username-input');
  const suggestionsBox = document.getElementById('username-suggestions');
  if (!input || !suggestionsBox) return;

  const showSuggestions = () => {
    suggestionsBox.style.visibility = 'visible';
    suggestionsBox.style.opacity = '1';
    suggestionsBox.style.transform = 'translateY(0)';
  };

  const hideSuggestions = () => {
    suggestionsBox.style.opacity = '0';
    suggestionsBox.style.transform = 'translateY(-10px)';
    setTimeout(() => { 
      if (suggestionsBox.style.opacity === '0') suggestionsBox.style.visibility = 'hidden'; 
    }, 200);
  };

  input.addEventListener('input', () => {
    const val = input.value.trim().toLowerCase();
    suggestionsBox.innerHTML = '';
    
    if (!val) {
      hideSuggestions();
      return;
    }

    const filtered = allTeachersForAutocomplete.filter(t => 
      t.username.toLowerCase().includes(val) || (t.name && t.name.toLowerCase().includes(val))
    );

    if (filtered.length === 0) {
      suggestionsBox.innerHTML = '<div style="padding: 10px; color: var(--text-muted); font-size: 0.85rem; text-align: center;">Tidak ditemukan</div>';
    } else {
      filtered.forEach(t => {
        const div = document.createElement('div');
        div.style.padding = '10px 15px';
        div.style.cursor = 'pointer';
        div.style.borderBottom = '1px solid var(--border-color)';
        div.style.transition = 'background 0.2s';
        div.innerHTML = `<div style="font-weight: 600; color: var(--text-main);">${t.username}</div><div style="font-size: 0.8rem; color: var(--text-muted);">${t.name || '-'}</div>`;
        
        div.onmouseenter = () => div.style.background = 'var(--bg-hover)';
        div.onmouseleave = () => div.style.background = 'transparent';
        
        div.onmousedown = (e) => {
          e.preventDefault(); // Mencegah input kehilangan fokus terlalu cepat
          input.value = t.username;
          hideSuggestions();
        };
        suggestionsBox.appendChild(div);
      });
    }
    showSuggestions();
  });

  input.addEventListener('focus', () => {
    if (input.value.trim() !== '') {
      input.dispatchEvent(new Event('input'));
    }
  });

  input.addEventListener('blur', () => {
    hideSuggestions();
  });
};

};


async function updateQuestionBankWorkspaceVisibility() {
  const workspace = document.getElementById('question-bank-workspace');
  const packetName = state.teacherSelectedPacket || '';
  const subjectId = state.teacherSelectedSubject || '';
  const classId = state.teacherSelectedClass || '';

  if (!packetName || !subjectId || !classId) {
    workspace.classList.add('hidden');
    return;
  }

  // All selected!
  workspace.classList.remove('hidden');

  // Update title to show context
  const workspaceTitle = document.getElementById('selected-workspace-title');
  const subjectSelect = document.getElementById('question-filter-subject');
  const classSelect = document.getElementById('question-filter-class');
  const subjectName = subjectSelect.options[subjectSelect.selectedIndex].text;
  const className = classSelect.options[classSelect.selectedIndex].text;
  
  const cleanClassName = className.replace(/^(kelas\s+)/i, '');
  workspaceTitle.dataset.subjectName = subjectName;
  workspaceTitle.dataset.className = cleanClassName;
  workspaceTitle.dataset.packetName = packetName;

  workspaceTitle.innerHTML = `${subjectName} <span style="font-weight: 300; opacity: 0.6; margin: 0 0.5rem;">/</span> <span style="font-size: 0.9em; font-weight: 500;">Kelas ${cleanClassName}</span>`;

  try {
    const list = await apiCall('/teacher/questions');
    const subjectQuestions = list.filter(q => q.subjectId === subjectId);
    renderTeacherQuestions(subjectQuestions);
  } catch (err) {
    console.error(err);
  }
}

function renderTeacherQuestions(subjectQuestions) {
  const selectedPacket = state.teacherSelectedPacket || 'Umum';
  const filteredQuestions = subjectQuestions.filter(q => q.paketSoal === selectedPacket);

  const workspaceTitle = document.getElementById('selected-workspace-title');
  const workspaceSubtitle = document.getElementById('selected-workspace-subtitle');
  
  if (workspaceSubtitle) {
    workspaceSubtitle.innerHTML = `
      <span class="badge badge-primary" style="margin-right: 0.25rem;"><i class="fa-solid fa-folder" style="margin-right: 0.25rem;"></i> Paket: ${selectedPacket}</span>
      <span class="badge badge-dark"><i class="fa-solid fa-file-circle-question" style="margin-right: 0.25rem;"></i> ${filteredQuestions.length} Soal</span>
    `;
  }

  // Store in cache for preview lookup
  state.loadedTeacherQuestions = filteredQuestions;

  const tbody = document.getElementById('table-questions-body');
  if (!tbody) return;
  tbody.innerHTML = '';

  if (filteredQuestions.length === 0) {
    tbody.innerHTML = `<tr><td colspan="8" class="text-center text-muted" style="padding: 2rem;">Belum ada butir soal untuk mata pelajaran/paket ini. Silakan buat baru atau impor soal CSV.</td></tr>`;
    return;
  }

  filteredQuestions.forEach((q, index) => {
    const visualStim = q.imageUrl ? `<a href="${resolveImageUrl(q.imageUrl)}" target="_blank" class="badge badge-primary"><i class="fa-solid fa-image"></i> Gambar</a>` : '-';
    let answersList = '';
    if (q.type === 'BS') {
      answersList = (q.correctAnswers || []).map((ans, idx) => `P${idx + 1}: ${ans}`).join(', ');
    } else if (q.type === 'PG' || q.type === 'PGK') {
      const labels = ['A', 'B', 'C', 'D', 'E', 'F', 'G', 'H'];
      answersList = (q.correctAnswers || []).map(ans => {
        const idx = (q.options || []).indexOf(ans);
        return idx !== -1 ? `${labels[idx]}. ${ans}` : ans;
      }).join(', ');
    } else if (q.type === 'Esai') {
      answersList = q.rambuJawaban ? `Rambu: ${q.rambuJawaban}` : 'Koreksi Manual';
    } else {
      answersList = (q.correctAnswers || []).join(', ');
    }
    
    // Escape quotes for safe title attribute
    const escapedText = (q.text || '').replace(/"/g, '&quot;');
    const escapedAnswers = answersList.replace(/"/g, '&quot;');
    
    tbody.innerHTML += `
      <tr>
        <td style="text-align: center;"><input type="checkbox" class="question-checkbox" value="${q.id}"></td>
        <td style="text-align: center; font-weight: 600;">${index + 1}</td>
        <td><span class="badge badge-purple">${getMapelSingkatan(q.subjectName)}</span></td>
        <td><span class="badge badge-dark">${q.type}</span></td>
        <td><div style="max-width: 260px; overflow: hidden; text-overflow: ellipsis; white-space: nowrap;" title="${escapedText}">${formatFractions(q.text)}</div></td>
        <td>${visualStim}</td>
        <td><div style="max-width: 200px; overflow: hidden; text-overflow: ellipsis; white-space: nowrap; color: var(--text-main);" title="${escapedAnswers}">${answersList}</div></td>
        <td>
          <div style="display: flex; gap: 5px; justify-content: flex-end;">
            <button class="btn btn-sm btn-outline-primary" onclick="previewQuestion('${q.id}')" title="Preview Soal"><i class="fa-solid fa-eye"></i></button>
            <button class="btn btn-sm btn-warning" onclick="editQuestion('${q.id}')" title="Ubah Soal"><i class="fa-solid fa-pen"></i></button>
            <button class="btn btn-sm btn-danger" onclick="deleteQuestion('${q.id}')" title="Hapus Soal"><i class="fa-solid fa-trash"></i></button>
          </div>
        </td>
      </tr>
    `;
  });

  initBulkDeleteHandlers();
}

function initBulkDeleteHandlers() {
  const selectAllCheckbox = document.getElementById('checkbox-select-all-questions');
  const checkboxes = document.querySelectorAll('.question-checkbox');
  let btnDeleteSelected = document.getElementById('btn-delete-selected-questions');

  if (!selectAllCheckbox || !btnDeleteSelected) return;

  // Ensure click listener is bound only once by replacing the node
  btnDeleteSelected.replaceWith(btnDeleteSelected.cloneNode(true));
  btnDeleteSelected = document.getElementById('btn-delete-selected-questions');

  const updateDeleteButtonVisibility = () => {
    const checkedCount = document.querySelectorAll('.question-checkbox:checked').length;
    btnDeleteSelected.style.display = checkedCount > 0 ? 'inline-block' : 'none';
    selectAllCheckbox.checked = checkedCount > 0 && checkedCount === checkboxes.length;
  };

  selectAllCheckbox.checked = false;
  btnDeleteSelected.style.display = 'none';

  selectAllCheckbox.addEventListener('change', (e) => {
    checkboxes.forEach(cb => {
      cb.checked = e.target.checked;
    });
    updateDeleteButtonVisibility();
  });

  checkboxes.forEach(cb => {
    cb.addEventListener('change', updateDeleteButtonVisibility);
  });
  
  btnDeleteSelected.addEventListener('click', async () => {
    const checkedBoxes = document.querySelectorAll('.question-checkbox:checked');
    const idsToDelete = Array.from(checkedBoxes).map(cb => cb.value);

    if (idsToDelete.length === 0) return;

    if (confirm(`Apakah Anda yakin ingin menghapus ${idsToDelete.length} soal yang dipilih? Aksi ini tidak dapat dibatalkan.`)) {
      try {
        const response = await apiCall('/teacher/questions/bulk-delete', 'POST', { ids: idsToDelete });
        showToast(response.message, 'success');
        loadTeacherQuestions();
      } catch (err) {
        showToast('Gagal menghapus soal terpilih.', 'danger');
      }
    }
  });
}

window.previewAllQuestions = function() {
  const questions = (state.loadedTeacherQuestions || []).slice();
  if (questions.length === 0) {
    showToast('Tidak ada soal untuk dipratinjau. Pilih mata pelajaran dan kelas terlebih dahulu.', 'error');
    return;
  }

  questions.sort((a, b) => (a.number || 0) - (b.number || 0));

  // Gather context info
  const subjectEl = document.getElementById('question-filter-subject');
  const classEl = document.getElementById('question-filter-class');
  const packetEl = document.getElementById('question-filter-packet');
  const subject = subjectEl ? subjectEl.value : '';
  const classId = classEl ? classEl.value : '';
  const packet = packetEl && packetEl.value ? packetEl.value : '';

  const infoText = [subject, classId ? ('Kelas ' + classId) : '', packet].filter(Boolean).join(' — ');
  document.getElementById('fpp-exam-info').textContent = infoText;
  document.getElementById('fpp-question-count').textContent = questions.length + ' soal';

  const labels = ['A', 'B', 'C', 'D', 'E', 'F', 'G', 'H'];
  let html = '';

  questions.forEach((q, idx) => {
    const correctAnswers = q.correctAnswers || [];
    let bodyHtml = '';

    // --- PG / PGK ---
    if (q.type === 'PG' || q.type === 'PGK') {
      const optionImages = q.optionImages || [];
      bodyHtml += '<ul class="report-opts">';
      (q.options || []).forEach((opt, oIdx) => {
        const isCorrect = correctAnswers.includes(opt) && opt !== '';
        const imgUrl = optionImages[oIdx];
        let optClass = 'report-opt';
        if (isCorrect) optClass += ' is-key';

        bodyHtml += '<li class="' + optClass + '">';
        bodyHtml += '<span class="report-opt-letter">' + labels[oIdx] + '.</span>';
        bodyHtml += '<div class="report-opt-content">' + formatFractions(opt);
        if (imgUrl) {
          bodyHtml += '<br><img class="report-q-img" src="' + resolveImageUrl(imgUrl) + '" style="max-height:80px; margin-top:4px;">';
        }
        bodyHtml += '</div>';
        
        if (isCorrect) {
          bodyHtml += '<span class="report-opt-tag key-tag"><i class="fa-solid fa-check"></i> Kunci Jawaban</span>';
        }
        bodyHtml += '</li>';
      });
      bodyHtml += '</ul>';

    // --- BS ---
    } else if (q.type === 'BS') {
      bodyHtml += '<table class="report-bs-table"><thead><tr>';
      bodyHtml += '<th>Pernyataan</th><th>Benar</th><th>Salah</th>';
      bodyHtml += '</tr></thead><tbody>';
      (q.options || []).forEach((opt, oIdx) => {
        const cv = correctAnswers[oIdx];
        bodyHtml += '<tr>';
        bodyHtml += '<td>' + formatFractions(opt) + '</td>';
        bodyHtml += '<td>' + (cv === 'Benar' ? '<span class="report-bs-check correct">✓ (Kunci)</span>' : '') + '</td>';
        bodyHtml += '<td>' + (cv === 'Salah' ? '<span class="report-bs-check correct">✓ (Kunci)</span>' : '') + '</td>';
        bodyHtml += '</tr>';
      });
      bodyHtml += '</tbody></table>';

    // --- Isian ---
    } else if (q.type === 'Isian') {
      bodyHtml += '<div class="fpp-isian-box"><strong>Kunci Jawaban:</strong> ' + (correctAnswers[0] || '-') + '</div>';

    // --- Esai ---
    } else if (q.type === 'Esai') {
      if (q.rambuJawaban) {
        bodyHtml += '<div class="fpp-rubric"><strong>Rambu Jawaban:</strong> ' + q.rambuJawaban + '</div>';
      }
    }

    html += '<div class="report-q">';
    html += '<div class="report-q-header">';
    html += '<span class="report-q-num">' + (idx + 1) + '.</span>';
    html += '<span class="report-q-badge">' + mapTypeLabel(q.type) + '</span>';
    html += '</div>';
    if (q.imageUrl) {
      html += '<img class="report-q-img" src="' + resolveImageUrl(q.imageUrl) + '">';
    }
    html += '<div class="report-q-text">' + formatFractions(q.text) + '</div>';
    html += bodyHtml;
    html += '</div>';
  });

  document.getElementById('fpp-content').innerHTML = html;
  document.getElementById('full-page-preview-container').style.display = 'block';
  document.getElementById('full-page-preview-container').scrollTop = 0;
};

async function openModalQuestionForm(qData = null) {
  try {
    // Populate subjects selector
    let subjects = await apiCall('/admin/subjects');
    if (state.role === 'guru') {
      const mappings = await apiCall('/admin/mappings');
      const teacherMappings = mappings.filter(m => m.teacherId === state.user.id);
      const mappedSubjectIds = [...new Set(teacherMappings.map(m => m.subjectId))];
      subjects = subjects.filter(s => mappedSubjectIds.includes(s.id));
    }
    const subjSelect = document.getElementById('question-subject-select');
    subjSelect.innerHTML = '';
    subjects.forEach(s => {
      subjSelect.innerHTML += `<option value="${s.id}">${s.name}</option>`;
    });

    const subjectId = qData ? qData.subjectId : (state.teacherSelectedSubject || '');
    if (subjectId) {
      const questions = await apiCall('/teacher/questions');
      const subjectQuestions = questions.filter(q => q.subjectId === subjectId);
      const uniquePackets = [...new Set(subjectQuestions.map(q => q.paketSoal).filter(Boolean))];
      const datalist = document.getElementById('existing-packets');
      if (datalist) {
        datalist.innerHTML = '';
        uniquePackets.forEach(p => {
          datalist.innerHTML += `<option value="${p}">`;
        });
      }
    }

    if (qData) {
      document.getElementById('question-id-hidden').value = qData.id;
      document.getElementById('question-subject-select').value = qData.subjectId;
      document.getElementById('question-subject-select').setAttribute('disabled', 'true');
      document.getElementById('question-type-select').value = qData.type;
      document.getElementById('question-image-url').value = qData.imageUrl || '';
      updateStimulusPreview(qData.imageUrl || '');
      const imgFileInput = document.getElementById('question-image-file');
      if (imgFileInput) imgFileInput.value = '';
      document.getElementById('question-text-input').value = qData.text;
      if (questionQuill) {
        questionQuill.clipboard.dangerouslyPasteHTML(qData.text);
      }
      document.getElementById('question-packet-input').value = qData.paketSoal || 'Umum';
      
      document.getElementById('modal-question-title').textContent = 'Ubah Butir Soal';
      renderOptionsBuilderFields(qData);
    } else {
      document.getElementById('question-id-hidden').value = '';
      document.getElementById('question-image-url').value = '';
      updateStimulusPreview('');
      const imgFileInput = document.getElementById('question-image-file');
      if (imgFileInput) imgFileInput.value = '';
      document.getElementById('question-text-input').value = '';
      if (questionQuill) {
        questionQuill.setText('');
      }
      document.getElementById('question-type-select').value = 'PG';
      document.getElementById('question-packet-input').value = (state.teacherSelectedPacket && state.teacherSelectedPacket !== 'ALL') ? state.teacherSelectedPacket : 'Umum';
      
      if (state.teacherSelectedSubject) {
        document.getElementById('question-subject-select').value = state.teacherSelectedSubject;
        document.getElementById('question-subject-select').setAttribute('disabled', 'true');
      } else {
        document.getElementById('question-subject-select').removeAttribute('disabled');
      }
      
      document.getElementById('modal-question-title').textContent = 'Buat Soal Baru';
      renderOptionsBuilderFields();
    }
    openModal('modal-question');
  } catch (err) {}
}

function renderOptionsBuilderFields(qData = null) {
  const type = document.getElementById('question-type-select').value;
  const container = document.getElementById('options-dynamic-fields');
  container.innerHTML = '';

  const data = qData && typeof qData.target === 'undefined' ? qData : null;

  if (type === 'PG' || type === 'PGK') {
    let options = data ? [...data.options] : ['', '', '', ''];
    let correct = data ? [...data.correctAnswers] : [];
    let optionImages = data && data.optionImages ? [...data.optionImages] : ['', '', '', ''];

    // Ensure at least 2 options
    while (options.length < 2) {
      options.push('');
      optionImages.push('');
    }

    const getOptionState = () => {
      const currentOpts = [];
      const currentImgs = [];
      const currentCorrect = [];
      
      const optionInputs = container.querySelectorAll('.pg-option-val');
      const imageInputs = container.querySelectorAll('.pg-option-img-val');
      
      optionInputs.forEach((inp, idx) => {
        currentOpts.push(inp.value.trim());
        currentImgs.push(imageInputs[idx] ? imageInputs[idx].value.trim() : '');
      });

      if (type === 'PG') {
        const selected = container.querySelector('input[name="pg-key"]:checked');
        if (selected) {
          const selectedIdx = parseInt(selected.value, 10);
          if (currentOpts[selectedIdx]) {
            currentCorrect.push(currentOpts[selectedIdx]);
          }
        }
      } else {
        const selected = container.querySelectorAll('input[name="pgk-key"]:checked');
        selected.forEach(chk => {
          const selectedIdx = parseInt(chk.value, 10);
          if (currentOpts[selectedIdx]) {
            currentCorrect.push(currentOpts[selectedIdx]);
          }
        });
      }
      return { options: currentOpts, optionImages: currentImgs, correctAnswers: currentCorrect };
    };

    const renderPGFields = () => {
      const labels = ['A', 'B', 'C', 'D', 'E', 'F', 'G', 'H'];
      const inputType = type === 'PG' ? 'radio' : 'checkbox';
      const inputName = type === 'PG' ? 'pg-key' : 'pgk-key';

      let html = `
        <div style="display: flex; justify-content: space-between; align-items: center; margin-bottom: 0.5rem;">
          <label style="font-weight: 600;">Pilihan Opsi & Kunci Jawaban</label>
          <button type="button" class="btn btn-sm btn-outline-primary" id="btn-add-pg-option" style="border: 1px solid var(--primary); border-radius: 6px; padding: 4px 8px; font-size: 0.8rem; background: transparent; cursor: pointer; color: var(--primary);">
            <i class="fa-solid fa-plus"></i> Tambah Pilihan
          </button>
        </div>
        <div id="pg-options-list-container" style="display: flex; flex-direction: column; gap: 8px;">
      `;

      options.forEach((opt, i) => {
        const checked = correct.includes(opt) && opt !== '';
        const imgUrl = optionImages[i] || '';

        html += `
          <div class="pg-option-row margin-top-sm" data-idx="${i}" style="display: flex; flex-direction: column; gap: 4px; padding: 8px; border-bottom: 1px dashed var(--border-color); background: rgba(0,0,0,0.01); border-radius: 6px;">
            <div class="flex-row align-center" style="gap: 8px;">
              <input type="${inputType}" name="${inputName}" value="${i}" ${checked ? 'checked' : ''}>
              <span class="bold" style="width: 20px;">${labels[i]}.</span>

async function handleQuestionSubmit(e) {
  e.preventDefault();
  const id = document.getElementById('question-id-hidden').value;
  const subjectId = document.getElementById('question-subject-select').value;
  const type = document.getElementById('question-type-select').value;
  let imageUrl = document.getElementById('question-image-url').value;
  
  const imageFileInput = document.getElementById('question-image-file');
  if (imageFileInput && imageFileInput.files.length > 0) {
    const formData = new FormData();
    formData.append('image', imageFileInput.files[0]);
    try {
      const uploadRes = await fetch(`${API_BASE}/upload-image`, {
        method: 'POST',
        headers: { 'Authorization': state.token },
        body: formData
      });
      if (!uploadRes.ok) throw new Error('Gagal mengunggah gambar dari komputer.');
      const data = await uploadRes.json();
      imageUrl = data.imageUrl;
    } catch(err) {
      showToast(err.message, 'error');
      return;
    }
  }

  const text = document.getElementById('question-text-input').value;

  let options = [];
  let correctAnswers = [];
  let optionImages = [];

  if (type === 'PG' || type === 'PGK') {
    const optionInputs = document.querySelectorAll('.pg-option-val');
    optionInputs.forEach(inp => {
      options.push(inp.value.trim());
    });

    const optionImageInputs = document.querySelectorAll('.pg-option-img-val');
    optionImageInputs.forEach(inp => {
      optionImages.push(inp.value.trim());
    });

    if (type === 'PG') {
      const selected = document.querySelector('input[name="pg-key"]:checked');
      if (!selected) {
        showToast('Pilih satu kunci jawaban untuk PG!', 'warning');
        return;
      }
      correctAnswers.push(options[parseInt(selected.value)]);
    } else {
      const selected = document.querySelectorAll('input[name="pgk-key"]:checked');
      if (selected.length === 0) {
        showToast('Pilih setidaknya satu kunci jawaban untuk PGK!', 'warning');
        return;
      }
      selected.forEach(chk => {
        correctAnswers.push(options[parseInt(chk.value)]);
      });
    }
  } else if (type === 'Isian') {
    const keyVal = document.getElementById('isian-correct-key').value.trim();
    if (!keyVal) {
      showToast('Kunci jawaban isian singkat harus diisi!', 'warning');
      return;
    }
    correctAnswers.push(keyVal);
  } else if (type === 'BS') {
    const rows = document.querySelectorAll('.bs-statement-row');
    rows.forEach(row => {
      const textVal = row.querySelector('.bs-statement-val').value.trim();
      const idx = parseInt(row.querySelector('.bs-statement-val').dataset.idx, 10);
      const checkedRadio = row.querySelector(`input[name="bs-correct-${idx}"]:checked`);
      options.push(textVal);
      correctAnswers.push(checkedRadio ? checkedRadio.value : 'Benar');
    });

    if (options.length === 0) {
      showToast('Minimal harus ada 1 pernyataan untuk tipe Benar/Salah!', 'warning');
      return;
    }
  }

  let rambuJawaban = '';
  const rubricInput = document.getElementById('question-rubric-input');
  if (rubricInput) {
    rambuJawaban = rubricInput.value.trim();
  }

  const paketSoal = document.getElementById('question-packet-input').value.trim() || 'Umum';
  const payload = {
    teacherId: state.user.id,
    subjectId,
    type,
    imageUrl,
    text,
    options,
    optionImages,
    correctAnswers,
    paketSoal,
    rambuJawaban
  };

  try {
    if (id) {
      await apiCall(`/teacher/questions/${id}`, 'PUT', payload);
      showToast('Soal ujian diperbarui.');
    } else {
      await apiCall('/teacher/questions', 'POST', payload);
      showToast('Soal baru berhasil ditambahkan.');
    }
    closeModal('modal-question');
    loadTeacherQuestions();
  } catch (err) {}
}

window.stopExamEmergency = async (id) => {
  if (confirm('Hentikan ujian ini secara darurat? Murid tidak akan dapat memasukkan token atau mengumpulkan jawaban lagi.')) {
    try {
      await apiCall(`/teacher/exams/${id}/emergency-stop`, 'POST');
      loadTeacherExams();
      showToast('Ujian dihentikan secara darurat.');
    } catch (err) {}
  }
};

window.deleteExam = async (id, title, isActive) => {
  if (isActive) {
    showToast('Ujian yang sedang aktif tidak bisa dihapus. Hentikan ujian terlebih dahulu!', 'warning');
    return;
  }
  if (!confirm(`Hapus ujian "${title}"?\n\nSemua data hasil ujian dan jawaban siswa untuk ujian ini juga akan ikut terhapus.\n\nKlik OK untuk melanjutkan.`)) return;

  try {
    await apiCall(`/teacher/exams/${id}`, 'DELETE');
    showToast(`Ujian "${title}" berhasil dihapus.`);
    loadTeacherExams();
  } catch (err) {}
};

window.editExam = async (id) => {
  try {
    const list = await apiCall('/teacher/exams');
    const ex = list.find(item => item.id === id);
    if (ex) {
      openModalExamForm(ex);
    }
  } catch (err) {}
};

async function updateExamClassSelect(subjectId, selectedClassVal = '') {
  const classSelect = document.getElementById('exam-class-select');
  if (!classSelect) return;
  let classes = await apiCall('/classes');
  if (state.role === 'guru') {
    const mappings = await apiCall('/admin/mappings');
    const teacherMappings = mappings.filter(m => m.teacherId === state.user.id);
    const classIds = [...new Set(teacherMappings.filter(m => m.subjectId === subjectId).map(m => m.classId))];
    classes = classes.filter(c => classIds.includes(c));
  }
  classSelect.innerHTML = '';
  classes.forEach(c => {
    classSelect.innerHTML += `<option value="${c}">${c}</option>`;
  });
  if (selectedClassVal) {
    classSelect.value = selectedClassVal;
  } else if (classSelect.options.length > 0) {
    classSelect.value = classSelect.options[0].value;
  }
}

async function generateUniqueToken(excludeExamId = null) {
  try {
    const list = await apiCall('/teacher/exams');
    const activeExams = list.filter(e => e.active && e.id !== excludeExamId);
    const activeTokens = activeExams.map(e => e.token.toUpperCase());
    
    let token = '';
    do {
      token = Math.random().toString(36).substring(2, 7).toUpperCase();
    } while (activeTokens.includes(token));
    
    return token;
  } catch (err) {
    return Math.random().toString(36).substring(2, 7).toUpperCase();
  }
}

async function openModalExamForm(exData = null) {
  try {
    let subjects = await apiCall('/admin/subjects');
    if (state.role === 'guru') {
      const mappings = await apiCall('/admin/mappings');
      const teacherMappings = mappings.filter(m => m.teacherId === state.user.id);
      const mappedSubjectIds = [...new Set(teacherMappings.map(m => m.subjectId))];
      subjects = subjects.filter(s => mappedSubjectIds.includes(s.id));
    }
    const subjSelect = document.getElementById('exam-subject-select');
    subjSelect.innerHTML = '';
    subjects.forEach(s => {
      subjSelect.innerHTML += `<option value="${s.id}">${s.name}</option>`;
    });

    const isEdit = exData && typeof exData.target === 'undefined';
    const form = document.getElementById('form-exam');

    if (isEdit) {
      document.getElementById('exam-id-hidden').value = exData.id;
      document.getElementById('exam-title-input').value = exData.title;
      document.getElementById('exam-subject-select').value = exData.subjectId;
      
      await updateExamClassSelect(exData.subjectId, exData.classId);
      
      document.getElementById('exam-date-input').value = exData.date;
      document.getElementById('exam-start-time').value = exData.startTime;
      document.getElementById('exam-end-time').value = exData.endTime;
      document.getElementById('exam-duration-input').value = exData.durationMinutes;
      document.getElementById('exam-shuffle-questions').checked = exData.shuffleQuestions !== false;
      
      const chkAntiCheat = document.getElementById('exam-anti-cheat-enabled');
      chkAntiCheat.checked = exData.antiCheatEnabled !== false;
      chkAntiCheat.dispatchEvent(new Event('change'));

      document.getElementById('exam-token-gen').value = exData.token;
      if (document.getElementById('exam-unlock-key')) {
        document.getElementById('exam-unlock-key').value = exData.unlockKey || '';
      }
      document.getElementById('modal-exam-title').textContent = 'Ubah Detail Ujian';
      
      await loadExamPackets(exData.subjectId, exData.paketSoal);
    } else {
      document.getElementById('exam-id-hidden').value = '';
      document.getElementById('exam-title-input').value = '';
      document.getElementById('exam-date-input').value = new Date().toISOString().split('T')[0];
      document.getElementById('exam-start-time').value = '08:00';
      document.getElementById('exam-end-time').value = '17:00';
      document.getElementById('exam-duration-input').value = '90';
      document.getElementById('exam-shuffle-questions').checked = true;
      
      const chkAntiCheat = document.getElementById('exam-anti-cheat-enabled');
      chkAntiCheat.checked = true;
      chkAntiCheat.dispatchEvent(new Event('change'));

      document.getElementById('exam-token-gen').value = await generateUniqueToken();
      if (document.getElementById('exam-unlock-key')) {
        document.getElementById('exam-unlock-key').value = Math.random().toString(36).substring(2, 7).toUpperCase();
      }
      document.getElementById('modal-exam-title').textContent = 'Jadwalkan Ujian Baru';
      
      const initialSubj = subjSelect.value;
      await updateExamClassSelect(initialSubj);
      if (initialSubj) {
        await loadExamPackets(initialSubj);
      }
    }

    openModal('modal-exam');
  } catch (err) {}
}

async function loadExamPackets(subjectId, selectedPacketVal = '') {
  const packetSelect = document.getElementById('exam-packet-select');
  if (!packetSelect) return;

  try {
    // Use the paket soal API to get structured packets for this subject
    const paketList = await apiCall(`/teacher/paket-soal?teacherId=${state.user.id}&subjectId=${subjectId}`);
    const packets = paketList.map(p => p.name);

    // Also add from existing questions as fallback (for older data)
    const questions = await apiCall('/teacher/questions');
    const subjectQuestions = questions.filter(q => q.subjectId === subjectId);
    const qPackets = [...new Set(subjectQuestions.map(q => q.paketSoal).filter(Boolean))];
    const allPackets = [...new Set([...packets, ...qPackets])];
    
    // Remove "Umum" if we don't want it explicitly added
    packetSelect.innerHTML = '';
    allPackets.forEach(p => {
      packetSelect.innerHTML += `<option value="${p}">${p}</option>`;
    });

    packetSelect.value = selectedPacketVal || (allPackets[0] || '');
  } catch (err) {
    console.error('Gagal memuat paket soal untuk ujian:', err);
    packetSelect.innerHTML = '';
    packetSelect.value = '';
  }
}

  const durationMinutes = document.getElementById('exam-duration-input').value;
  const shuffleQuestions = document.getElementById('exam-shuffle-questions').checked;
  const antiCheatEnabled = document.getElementById('exam-anti-cheat-enabled').checked;
  const token = document.getElementById('exam-token-gen').value.trim();
  const paketSoal = document.getElementById('exam-packet-select').value || 'Umum';
  const unlockKey = document.getElementById('exam-unlock-key') ? document.getElementById('exam-unlock-key').value.trim().toUpperCase() : '';

  const payload = {
    title, subjectId, classId, token, date, startTime, endTime, durationMinutes, shuffleQuestions, antiCheatEnabled, teacherId: state.user.id, paketSoal, unlockKey
  };

  try {
    if (id) {
      await apiCall(`/teacher/exams/${id}`, 'PUT', { ...payload, active: true });
      showToast('Ujian berhasil di-update.');
    } else {
      await apiCall('/teacher/exams', 'POST', payload);
      showToast('Ujian baru dirilis.');
    }
    closeModal('modal-exam');
    loadTeacherExams();
  } catch (err) {}
}

async function loadLiveMonitoringData() {
  try {
    const data = await apiCall('/teacher/monitoring');
    const tbody = document.getElementById('table-monitoring-body');
    tbody.innerHTML = '';
    
    let displayData = data;
    if (state.hideFinishedMonitoring) {
      displayData = data.filter(att => att.status !== 'finished');

      tbody.innerHTML = '<tr><td colspan="6" class="text-center text-muted">Belum ada murid aktif di ruang ujian.</td></tr>';

    }

    displayData.forEach(att => {

        : '<span class="badge badge-success">0 kali (Fokus)</span>';

        </tr>
      `;

async function loadTeacherAttempts() {
  try {
    const exams = await apiCall('/teacher/exams?includeDeleted=true');
    const examFilter = document.getElementById('grading-exam-filter');
    if (!examFilter) return;

    // Remember currently selected exam value
    const selectedExamId = examFilter.value;

    examFilter.innerHTML = '<option value="" disabled selected>Pilih Jadwal Ujian</option>';
    exams.forEach(e => {
      let formattedDate = '';
      if (e.date) {
        const parts = e.date.split('-');
        if (parts.length === 3) {
          formattedDate = `${parts[2]}/${parts[1]}/${parts[0]}`;
        } else {
          formattedDate = e.date;
        }
      }
      const dateSuffix = formattedDate ? ` - ${formattedDate}` : '';
      examFilter.innerHTML += `<option value="${e.id}">${e.title}${dateSuffix} (${e.subjectName} - ${e.classId})</option>`;
    });

    if (selectedExamId && exams.some(e => e.id === selectedExamId)) {
      examFilter.value = selectedExamId;
    }

    if (!examFilter.dataset.listenerBound) {
      examFilter.dataset.listenerBound = 'true';
      examFilter.addEventListener('change', () => {
        renderFilteredAttempts();
      });
    }

    const btnSaveWeights = document.getElementById('btn-save-weights');
    if (btnSaveWeights && !btnSaveWeights.dataset.listenerBound) {
      btnSaveWeights.dataset.listenerBound = 'true';
      btnSaveWeights.addEventListener('click', async () => {
        const examId = examFilter.value;
        if (!examId) return showToast('Pilih ujian terlebih dahulu.', 'danger');

        const weights = {
          PG: parseInt(document.getElementById('weight-pg').value, 10) || 0,
          PGK: parseInt(document.getElementById('weight-pgk').value, 10) || 0,
          BS: parseInt(document.getElementById('weight-bs').value, 10) || 0,
          Isian: parseInt(document.getElementById('weight-isian').value, 10) || 0,
          Esai: parseInt(document.getElementById('weight-esai').value, 10) || 0
        };

        btnSaveWeights.disabled = true;
        btnSaveWeights.innerHTML = '<i class="fa-solid fa-spinner fa-spin"></i> Menyimpan...';
        try {
          await apiCall(`/teacher/exams/${examId}/weights`, 'POST', { weights });
          showToast('Bobot soal berhasil disimpan.');
          renderFilteredAttempts();
        } catch (err) {
          showToast('Gagal menyimpan bobot.', 'danger');
        } finally {
          btnSaveWeights.disabled = false;
          btnSaveWeights.innerHTML = '<i class="fa-solid fa-check"></i> Simpan Bobot';
        }
      });
    }

    async function renderFilteredAttempts() {
      const examId = examFilter.value;
      const tbody = document.getElementById('table-attempts-body');
      const weightsContainer = document.getElementById('grading-weights-container');
      const attemptsCard = document.getElementById('grading-attempts-card');
      
      if (!tbody) return;

      const btnDownloadItemAnalysis = document.getElementById('btn-download-item-analysis');

      if (!examId) {
        tbody.innerHTML = '<tr><td colspan="6" class="text-center text-muted" style="padding: 2rem;">Silakan pilih jadwal ujian terlebih dahulu.</td></tr>';
        if (btnDownloadItemAnalysis) btnDownloadItemAnalysis.style.display = 'none';
        if (weightsContainer) weightsContainer.style.display = 'none';
        if (attemptsCard) attemptsCard.style.display = 'block';
        return;
      }
      
      const exams = await apiCall('/teacher/exams?includeDeleted=true');
      const exam = exams.find(e => e.id === examId);
      if (exam) {
        if (weightsContainer) {
          weightsContainer.style.display = 'block';
          document.getElementById('weight-pg').value = exam.weights?.PG ?? 1;
          document.getElementById('weight-pgk').value = exam.weights?.PGK ?? 2;
          document.getElementById('weight-bs').value = exam.weights?.BS ?? 2;
          document.getElementById('weight-isian').value = exam.weights?.Isian ?? 2;
          document.getElementById('weight-esai').value = exam.weights?.Esai ?? 3;
        }
        
        if (!exam.weights) {
          // If weights are not set, hide the attempts table and wait for weights
          if (attemptsCard) attemptsCard.style.display = 'none';
          if (btnDownloadItemAnalysis) btnDownloadItemAnalysis.style.display = 'none';
          return; // Stop here, don't fetch attempts
        } else {
          if (attemptsCard) attemptsCard.style.display = 'block';
        }
      }

      if (btnDownloadItemAnalysis) {
        const token = localStorage.getItem('cbt_token') || '';
        btnDownloadItemAnalysis.href = `/api/teacher/reports/${examId}/item-analysis?token=${token}`;
        btnDownloadItemAnalysis.style.display = 'flex';
      }

      tbody.innerHTML = '<tr><td colspan="6" class="text-center text-muted" style="padding: 2rem;"><i class="fa-solid fa-spinner fa-spin"></i> Memuat data jawaban siswa...</td></tr>';

      try {
        const list = await apiCall('/teacher/attempts');
        const filteredList = list.filter(att => att.examId === examId);
        filteredList.sort((a, b) => (a.studentName || '').localeCompare(b.studentName || '', undefined, { numeric: true, sensitivity: 'base' }));
        window.currentGradingAttemptsList = filteredList;

        tbody.innerHTML = '';
        if (filteredList.length === 0) {
          tbody.innerHTML = '<tr><td colspan="6" class="text-center text-muted" style="padding: 2rem;">Belum ada siswa yang memulai ujian ini.</td></tr>';
          return;
        }

        filteredList.forEach(att => {
          let scoreLabel = '';
          let gradeBtn = '';
          let pdfDownloadBtn = '';

          if (att.status === 'finished') {
            const token = localStorage.getItem('cbt_token') || '';
            scoreLabel = att.score !== null ? `<strong>${att.score}</strong>` : '<span class="badge badge-warning">Butuh Koreksi</span>';
            gradeBtn = `<button class="btn btn-sm btn-primary" onclick="openGradingModal('${att.id}', '${att.studentName}', '${att.examTitle}')"><i class="fa-solid fa-file-signature"></i> Nilai</button>`;
            pdfDownloadBtn = `<button class="btn btn-sm btn-danger" onclick="openStudentReport('${att.id}')"><i class="fa-solid fa-file-pdf"></i> PDF</button>`;
          } else {
            scoreLabel = '<span class="badge badge-info" style="background: rgba(13, 202, 240, 0.15); color: #0dcaf0; border: 1px solid rgba(13, 202, 240, 0.3); padding: 5px 8px; border-radius: 4px;"><i class="fa-solid fa-spinner fa-spin"></i> Sedang Mengerjakan</span>';
            gradeBtn = '<span class="font-sm text-muted">Sedang Mengerjakan</span>';
            pdfDownloadBtn = '-';
          }

          tbody.innerHTML += `
            <tr>
              <td>${att.studentName}</td>
              <td><span class="badge badge-dark">${att.classId}</span></td>
              <td>${att.examTitle}</td>
              <td>${gradeBtn}</td>
              <td>${scoreLabel}</td>
              <td>${pdfDownloadBtn}</td>
            </tr>
          `;
        });
      } catch (err) {
        tbody.innerHTML = '<tr><td colspan="6" class="text-center text-danger" style="padding: 2rem;">Gagal memuat data jawaban siswa.</td></tr>';
      }
    }

    await renderFilteredAttempts();
  } catch (err) {
    console.error('Error loading grading screen:', err);
  }
}

window.openStudentReport = async (attemptId) => {
  try {
    const list = await apiCall('/teacher/attempts');
    const att = list.find(a => a.id === attemptId);
    if (!att) return showToast('Data attempt tidak ditemukan', 'error');

    const exam = (await apiCall('/teacher/exams?includeDeleted=true')).find(e => e.id === att.examId);
    if (!exam) return showToast('Data ujian tidak ditemukan', 'error');

    const questions = await apiCall('/teacher/questions');
    let subjectQuestions = questions.filter(q => {
      const matchSubject = q.subjectId === exam.subjectId;
      const matchPacket = !exam.paketSoal || q.paketSoal === exam.paketSoal;
      return matchSubject && matchPacket;
    });
    // Sort questions by number
    subjectQuestions.sort((a, b) => (a.number || 0) - (b.number || 0));

    const settings = await apiCall('/settings');
    const schoolName = settings.schoolName || 'Sekolah Dasar Negeri';
    const academicYear = settings.academicYear || '2023/2024';
    const logoUrl = settings.schoolLogo ? resolveImageUrl(settings.schoolLogo) : 'assets/logo.png';

    // Info header
    const examInfoText = `${att.studentName} — ${exam.title}`;
    document.getElementById('report-student-info').textContent = examInfoText;
    
    // Store filename for printing
    const subjectNameForFilename = att.subjectName || '-';
    state.currentReportFilename = `${att.studentName || 'Siswa'} - ${exam.title || 'Ujian'} - ${subjectNameForFilename}`;

    // Build the Paper HTML
    let html = '';

    // Kop Surat
    html += `
      <div class="report-kop">
        <img src="${logoUrl}" alt="Logo Sekolah" class="report-kop-logo" onerror="this.src='assets/logo.png'">
        <div class="report-kop-text">
          <h2>${schoolName}</h2>
          <p>Tahun Pelajaran ${academicYear}</p>
          <h3 style="margin: 3px 0 0 0; font-size: 0.9rem; font-weight: 600; color: hsl(225, 25%, 35%);">${exam.title}</h3>
        </div>
        ${att.teacherNote ? `<div style="margin-right: 100px; max-width: 200px; font-family: 'Caveat', cursive; font-weight: 600; color: #0056b3; font-size: 1.1rem; line-height: 1.2; transform: rotate(-3deg);">${att.teacherNote.replace(/\n/g, '<br>')}</div>` : ''}
        <div class="report-score-wrap">
          <svg class="report-score-svg" viewBox="0 0 100 100" preserveAspectRatio="none">
            <path d="M30,25 C45,10 75,10 85,30 C95,50 85,85 50,90 C15,95 5,55 15,30 C25,5 55,5 70,20" 
                  fill="none" stroke="hsl(355, 80%, 45%)" stroke-width="2.5" stroke-linecap="round" stroke-linejoin="round"
                  vector-effect="non-scaling-stroke"
                  style="opacity: 0.85; transform-origin: center; transform: scale(0.95) rotate(-2deg);" />
          </svg>
          <div class="report-score-value">${att.score !== null ? Math.round(att.score) : '?'}</div>
        </div>
      </div>
    `;

    // Identity
    const subjectName = att.subjectName || '-';
    html += `
      <div class="report-identity">
        <table>
          <tr><td>Nama Siswa</td><td>:</td><td><strong>${att.studentName || '-'}</strong></td></tr>
          <tr><td>NIS / NISN</td><td>:</td><td>${att.studentNis || '-'} / ${att.studentNisn || '-'}</td></tr>
          <tr><td>Kelas</td><td>:</td><td>${att.classId || '-'}</td></tr>
          <tr><td>Mata Pelajaran</td><td>:</td><td>${subjectName}</td></tr>
          <tr><td>Judul Ujian</td><td>:</td><td>${exam.title || '-'}</td></tr>
        </table>
      </div>
    `;

    // Questions
    const weights = exam.weights || { PG: 1, PGK: 2, Isian: 2, Esai: 3, BS: 2 };
    let totalEarnedPoints = 0;
    let totalMaxPoints = 0;

    subjectQuestions.forEach((q, index) => {
      const isEssay = q.type === 'Esai';
      const isIsian = q.type === 'Isian';
      const isPG = q.type === 'PG';
      const isPGK = q.type === 'PGK';
      const isBS = q.type === 'BS';

      const studentAns = att.answers[q.id];
      let studentAnsEmpty = false;
      if (studentAns === undefined || studentAns === null || studentAns === '' || (Array.isArray(studentAns) && !studentAns.some(val => val !== undefined && val !== null && val !== ''))) {
        studentAnsEmpty = true;
      }

      let qWeight = 0;
      if (isPG) qWeight = parseInt(weights.PG, 10) || 0;
      else if (isPGK) qWeight = parseInt(weights.PGK, 10) || 0;
      else if (isBS) qWeight = parseInt(weights.BS, 10) || 0;
      else if (isIsian) qWeight = parseInt(weights.Isian, 10) || 0;
      else if (isEssay) qWeight = parseInt(weights.Esai, 10) || 0;

      let earnedPoints = 0;
      if (att.essayGrades && att.essayGrades[q.id] !== undefined && att.essayGrades[q.id] !== null && att.essayGrades[q.id] !== '') {
        earnedPoints = parseFloat(att.essayGrades[q.id]);
      } else if (isPG) {
        const stuAnsStr = Array.isArray(studentAns) ? studentAns[0] : studentAns;
        if (!studentAnsEmpty && q.correctAnswers && q.correctAnswers[0] === stuAnsStr) {
          earnedPoints = qWeight;
        }
      } else if (isPGK) {
        const stuArr = Array.isArray(studentAns) ? studentAns : [];
        const correctArr = q.correctAnswers || [];
        let matchCount = 0; let wrongCount = 0;
        stuArr.forEach(ans => {
          if (correctArr.includes(ans)) matchCount++;
          else wrongCount++;
        });
        const totalOptionsCount = (q.options || []).length;
        let earnedFraction = 0;
        if (stuArr.length === totalOptionsCount || matchCount === 0) earnedFraction = 0;
        else if (matchCount === correctArr.length && wrongCount === 0) earnedFraction = 1;
        else if (matchCount >= 1) earnedFraction = 0.5;
        earnedPoints = qWeight * earnedFraction;
      } else if (isBS) {
        const correctArr = q.correctAnswers || [];
        const stuArr = Array.isArray(studentAns) ? studentAns : [];
        const totalStatements = (q.options || []).length;
        let matchCount = 0;
        correctArr.forEach((correctVal, index) => {
          if (stuArr[index] === correctVal && correctVal !== '') matchCount++;
        });
        const earnedFraction = totalStatements > 0 ? (matchCount / totalStatements) : 0;
        earnedPoints = qWeight * earnedFraction;
      } else if (isIsian) {
        const stuAnsStr = studentAns ? studentAns.toString().toLowerCase() : '';
        if (!studentAnsEmpty && q.correctAnswers && q.correctAnswers.some(k => k.toLowerCase() === stuAnsStr)) {
          earnedPoints = qWeight;
        }
      }
      totalEarnedPoints += earnedPoints;
      totalMaxPoints += qWeight;
      const displayPoints = '+' + (Math.round(earnedPoints * 100) / 100);

      html += `<div class="report-q">`;
      html += `<div class="report-q-header">`;
      html += `<span class="report-q-num">${index + 1}.</span>`;
      html += `<span class="report-q-badge">${mapTypeLabel(q.type)}</span>`;
      html += `<span class="report-q-score-stamp" style="font-family: 'Just Me Again Down Here', cursive; font-size: 1.4rem; font-weight: 400; color: hsl(355, 80%, 45%); margin-left: auto;">${displayPoints}</span>`;
      html += `</div>`;
      if (q.imageUrl) {
        html += `<img class="report-q-img" src="${resolveImageUrl(q.imageUrl)}">`;
      }
      html += `<div class="report-q-text">${formatFractions(q.text)}</div>`;

      // PG / PGK
      if (isPG || isPGK) {
        const labels = ['A', 'B', 'C', 'D', 'E', 'F', 'G', 'H'];
        const correctAnswers = q.correctAnswers || [];
        const stuArr = Array.isArray(studentAns) ? studentAns : [studentAns];
        
        html += `<ul class="report-opts">`;
        (q.options || []).forEach((opt, oIdx) => {
          const isKey = correctAnswers.includes(opt) && opt !== '';
          const isStu = !studentAnsEmpty && stuArr.includes(opt) && opt !== '';
          const isStudentWrong = isStu && !isKey;
          
          let optClass = 'report-opt';
          if (isKey) optClass += ' is-key';
          if (isStudentWrong) optClass += ' is-student-wrong';

          html += `<li class="${optClass}">`;
          html += `<span class="report-opt-letter">${labels[oIdx]}.</span>`;
          html += `<div class="report-opt-content">${formatFractions(opt)}`;
          if (q.optionImages && q.optionImages[oIdx]) {
            html += `<br><img class="report-q-img" src="${resolveImageUrl(q.optionImages[oIdx])}" style="max-height:80px; margin-top:4px;">`;
          }
          html += `</div>`;
          
          if (isKey && isStu) {
            html += `<span class="report-opt-tag key-tag"><i class="fa-solid fa-check"></i> Jawaban Benar</span>`;
          } else if (isKey && !isStu) {

          }

        });

      }
      // Benar Salah
      else if (isBS) {

        html += `<th>Pernyataan</th><th>Benar</th><th>Salah</th><th>Status</th>`;

          const stuVal = stuArr[idx] || '';

          html += `<tr>`;
          html += `<td>${formatFractions(opt)}</td>`;
          
          // Col Benar
          html += `<td>`;
          if (stuVal === 'Benar' && keyVal === 'Benar') html += `<span class="report-bs-check correct">✓</span>`;
          else if (stuVal === 'Benar') html += `<span class="report-bs-check wrong">✗ (Siswa)</span>`;
          else if (keyVal === 'Benar') html += `<span class="report-bs-check correct">✓ (Kunci)</span>`;
          html += `</td>`;
          
          // Col Salah
          html += `<td>`;
          if (stuVal === 'Salah' && keyVal === 'Salah') html += `<span class="report-bs-check correct">✓</span>`;
          else if (stuVal === 'Salah') html += `<span class="report-bs-check wrong">✗ (Siswa)</span>`;
          else if (keyVal === 'Salah') html += `<span class="report-bs-check correct">✓ (Kunci)</span>`;
          html += `</td>`;
          
          // Col Status
          html += `<td>`;
          if (stuVal === '') html += `<span class="text-muted" style="font-size:0.75rem;">Kosong</span>`;
          else if (isMatch) html += `<span style="color:var(--success); font-weight:bold;">Benar</span>`;
          else html += `<span style="color:var(--danger); font-weight:bold;">Salah</span>`;
          html += `</td>`;
          
          html += `</tr>`;
        });
        html += `</tbody></table>`;
      }
      // Isian Singkat
      else if (isIsian) {
        const keyAns = (q.correctAnswers && q.correctAnswers[0]) ? q.correctAnswers[0].toLowerCase() : '';
        const stuAnsStr = studentAns ? studentAns.toString().toLowerCase() : '';

           isMatch = q.correctAnswers.some(k => k.toLowerCase() === stuAnsStr);

          html += `<i class="fa-solid fa-xmark" style="margin-top:2px;"></i>`;

function promptGeminiApiKey(force = false) {
  return new Promise((resolve) => {
    let geminiApiKey = localStorage.getItem('gemini_api_key');
    const container = document.getElementById('ai-api-key-container');
    const input = document.getElementById('ai-api-key-input');
    const btnSave = document.getElementById('btn-save-api-key');
    const btnClose = document.getElementById('btn-close-api-key');
    const btnToggle = document.getElementById('btn-toggle-api-key');

    if (!geminiApiKey || force) {
      input.value = geminiApiKey || '';
      input.type = 'password';
      if (btnToggle) {
        btnToggle.className = 'fa-solid fa-eye';
        btnToggle.onclick = () => {
          if (input.type === 'password') {
            input.type = 'text';
            btnToggle.className = 'fa-solid fa-eye-slash';
          } else {
            input.type = 'password';
            btnToggle.className = 'fa-solid fa-eye';
          }
        };
      }
      container.style.display = 'block';
      
      const cleanup = () => {
        btnSave.onclick = null;
        btnClose.onclick = null;
        if (btnToggle) btnToggle.onclick = null;
        container.style.display = 'none';
      };

      btnSave.onclick = () => {
        const val = input.value.trim();
        if (val) {
          localStorage.setItem('gemini_api_key', val);
          cleanup();
          if (force) alert('API Key Gemini berhasil disimpan!');
          resolve(val);
        } else {
          alert('API Key tidak boleh kosong!');
        }
      };

      btnClose.onclick = () => {
        cleanup();
        resolve(null);
      };
    } else {
      resolve(geminiApiKey);
    }
  });
}

async function handleAIGradeAll(essayQuestions, studentAnswers) {
  let geminiApiKey = await promptGeminiApiKey();
  if (!geminiApiKey) {
    return; // Dibatalkan pengguna
  }

  console.log("handleAIGradeAll triggered", essayQuestions, studentAnswers);
  const btnAiGrade = document.getElementById('btn-ai-grade-all');
  btnAiGrade.disabled = true;
  btnAiGrade.innerHTML = '<i class="fa-solid fa-spinner fa-spin"></i> Sedang Menilai...';

  // Process sequentially to not overload the API
  for (const eq of essayQuestions) {
    const studentAns = studentAnswers[eq.id];
    const panel = document.getElementById(`ai-panel-${eq.id}`);
    
    if (!studentAns) {
      panel.style.display = 'block';
      panel.innerHTML = '<div class="text-muted"><i class="fa-solid fa-circle-info"></i> Tidak dapat dinilai AI karena jawaban kosong.</div>';
      continue;
    }

    panel.style.display = 'block';
    panel.innerHTML = '<div class="text-center text-muted"><i class="fa-solid fa-robot fa-bounce"></i> AI sedang membaca jawaban...</div>';
    panel.className = 'ai-grade-panel loading';

    try {
      // Find reference answer if any
      const referenceAns = (eq.correctAnswers && eq.correctAnswers.length > 0) ? eq.correctAnswers.join(', ') : (eq.rambuJawaban || '');
      
      const gradeInput = document.getElementById(`grade-input-${eq.id}`);
      const dynamicMaxScore = gradeInput ? parseInt(gradeInput.getAttribute('max'), 10) : 100;
      
      console.log(`Sending to AI for question: ${eq.id}`);
      const res = await fetch(`${API_BASE}/teacher/ai/grade-essay`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${localStorage.getItem('cbt_token')}`
        },
        body: JSON.stringify({
          question: eq.text,
          studentAnswer: studentAns,
          referenceAnswer: referenceAns,
          maxScore: dynamicMaxScore,
          apiKey: geminiApiKey
        })
      });

      const data = await res.json();
      console.log(`AI response for ${eq.id}:`, data);
      panel.className = 'ai-grade-panel'; // Remove loading class

      if (!res.ok || !data.success) {
        panel.innerHTML = `<div class="text-danger"><i class="fa-solid fa-triangle-exclamation"></i> Gagal menilai: ${data.message || data.error}</div>`;
        if (data.message && data.message.includes('API Key')) {
            localStorage.removeItem('gemini_api_key');
        }
        continue;
      }

      const grade = data.grading;
      
      // Langsung terapkan skor ke input
      window.applyAiScore(eq.id, grade.score);

      panel.innerHTML = `
        <div class="ai-grade-header" style="margin-bottom: 0;">
          <div class="ai-grade-title" style="font-size: 0.85em;"><i class="fa-solid fa-check-circle" style="color: var(--success);"></i> Berhasil Dinilai AI</div>
        </div>
        <details style="margin-top: 5px; font-size: 0.8em; color: var(--text-muted);">
          <summary style="cursor: pointer; opacity: 0.8;">Lihat Analisis Singkat</summary>
          <div style="padding-top: 5px; line-height: 1.3; font-style: italic;">
            ${grade.feedback}
          </div>
        </details>
      `;

    } catch (err) {
      console.error(`AI error for ${eq.id}:`, err);
      panel.className = 'ai-grade-panel';
      panel.innerHTML = '<div class="text-danger"><i class="fa-solid fa-triangle-exclamation"></i> Gagal terhubung ke AI.</div>';
    }
  }

  btnAiGrade.disabled = false;
  btnAiGrade.innerHTML = '<i class="fa-solid fa-robot"></i> Koreksi Semua dengan AI';
}

window.applyAiScore = function(questionId, score) {
  const input = document.getElementById(`grade-input-${questionId}`);
  if (input) {
    input.value = score;
    // Highlight input momentarily to show it updated
    input.style.backgroundColor = 'var(--success-light)';
    setTimeout(() => { input.style.backgroundColor = ''; }, 1000);
  }
};

    }
    essayGrades[qId] = val;
  });

  if (!valid) return;

  const noteInput = document.getElementById('teacher-note-input');
  const teacherNote = noteInput ? noteInput.value.trim() : '';

  try {
    await apiCall(`/teacher/attempts/${attemptId}/grade`, 'POST', { essayGrades, teacherNote });
    closeModal('modal-grade-essay');
    loadTeacherAttempts();
    showToast('Penilaian esai tersimpan dan nilai terupdate.');
  } catch (err) {}
}

async function handleStudentTokenVerify() {
  const token = document.getElementById('exam-token-input').value.trim();

  if (!token) {
    showToast('Silakan masukkan token ujian terlebih dahulu.', 'warning');
    return;
  }

  // Automatically request fullscreen if anti-cheat is enabled
  if ((!state.settings || state.settings.antiCheatEnabled !== false) && (!state.currentExam || state.currentExam.antiCheatEnabled !== false)) {
    if (document.documentElement.requestFullscreen) {
      document.documentElement.requestFullscreen().catch(() => {});
    }
  }

  try {
    const res = await apiCall('/student/verify-token', 'POST', { token, classId: state.user.classId });
    showToast('Token Ujian valid! Membuka ruang ujian...');
    
    // Save to local cache
    state.currentExam = res;
    localStorage.setItem('cbt_active_exam_data', JSON.stringify(res));

    // Clear verify panel
    document.getElementById('exam-token-verification').classList.add('hidden');
    document.getElementById('exam-main-interface').classList.remove('hidden');

    // Show exam header
    const examHeader = document.querySelector('.exam-header');
    if (examHeader) examHeader.classList.remove('hidden');

    // Trigger exam loading logic
    startExamLogic(res.examId);
  } catch (err) {}
}

async function startExamLogic(examId, isResume = false) {
  try {
    const res = await apiCall('/student/exam/start', 'POST', { examId, nis: state.user.nis, nisn: state.user.nisn });
    
    state.activeAttemptId = res.attempt.id;
    localStorage.setItem('cbt_active_attempt_id', res.attempt.id);

    state.questions = res.questions;
    state.studentAnswers = res.attempt.answers || {};
    state.activeQuestionIndex = 0;

    // Load active title and logo
    document.getElementById('student-exam-title').textContent = state.currentExam.title;
    document.getElementById('student-exam-subject').textContent = state.currentExam.subjectName;
    const logoEl = document.getElementById('student-exam-school-logo');
    if (logoEl) logoEl.src = state.settings?.schoolLogo || 'img/tutwuri.png';

    // Renders
    renderQuestionGrid();
    renderQuestionContent();

    // Set remaining time and start counter clock timer
    state.remainingTimeSeconds = res.attempt.remainingTimeSeconds !== undefined ? res.attempt.remainingTimeSeconds : state.currentExam.duration * 60;
    startExamTimer();

    // Start autosave trigger (autosave every 10 seconds)
    setupAutosaveTimer();

    // Show exam header
    const examHeader = document.querySelector('.exam-header');
    if (examHeader) examHeader.classList.remove('hidden');

    document.getElementById('exam-token-verification').classList.add('hidden');
    document.getElementById('exam-main-interface').classList.remove('hidden');

    // Anti-cheat: check if locked
    if ((!state.settings || state.settings.antiCheatEnabled !== false) && (!state.currentExam || state.currentExam.antiCheatEnabled !== false)) {
      if (res.attempt.isLocked) {
        const overlay = document.getElementById('cheat-overlay');
        if (overlay) overlay.classList.remove('hidden');
      }
    }
  } catch (err) {
    if (state.token) {
      // Don't log out completely unless it's a 401 (handled by apiCall). Just clear active exam.
      localStorage.removeItem('cbt_active_exam_data');
      localStorage.removeItem('cbt_active_attempt_id');
      state.currentExam = null;
      state.activeAttemptId = null;
    }
    loadDataAndRoute();
  }
}

function renderQuestionGrid() {
  const grid = document.getElementById('exam-questions-grid');
  grid.innerHTML = '';

  let html = '';

  state.questions.forEach((q, index) => {
    const answer = state.studentAnswers[q.id];
    const isAnswered = answer && (Array.isArray(answer) ? answer.some(val => val !== undefined && val !== null && val !== '') : answer !== '');
    const isDoubt = state.studentDoubtQuestions[q.id];

    let extraClass = '';
    if (isDoubt) extraClass = 'doubt';
    else if (isAnswered) extraClass = 'answered';

    if (index === state.activeQuestionIndex) extraClass += ' active';

    html += `
      <div class="grid-box ${extraClass}" onclick="jumpToQuestion(${index})">${index + 1}</div>
    `;
  });

  grid.innerHTML = html;
}

window.selectOption = (optionIdx) => {
  const currentQ = state.questions[state.activeQuestionIndex];
  const optionVal = currentQ.options[optionIdx];
  let currentAns = state.studentAnswers[currentQ.id];

  if (currentQ.type === 'PG') {
    state.studentAnswers[currentQ.id] = [optionVal];
  } else if (currentQ.type === 'PGK') {
    if (!Array.isArray(currentAns)) currentAns = [];
    if (currentAns.includes(optionVal)) {
      currentAns = currentAns.filter(item => item !== optionVal);
    } else {
      currentAns.push(optionVal);
    }
    state.studentAnswers[currentQ.id] = currentAns;
  }

  // Trigger quick local save
  saveStudentAnswersToBackend();

  renderQuestionContent();
  renderQuestionGrid();
};

window.handleIsianInput = (val) => {
  const currentQ = state.questions[state.activeQuestionIndex];
  state.studentAnswers[currentQ.id] = val;
  renderQuestionGrid();
};

window.handleEssayInput = (val) => {
  const currentQ = state.questions[state.activeQuestionIndex];
  state.studentAnswers[currentQ.id] = val;
  renderQuestionGrid();
};

window.selectBSAnswer = (statementIdx, value) => {
  const currentQ = state.questions[state.activeQuestionIndex];
  let currentAns = state.studentAnswers[currentQ.id];

  if (!Array.isArray(currentAns)) {
    currentAns = Array(currentQ.options.length).fill('');
  }

  currentAns[statementIdx] = value;
  state.studentAnswers[currentQ.id] = currentAns;

  saveStudentAnswersToBackend();
  renderQuestionGrid();
};

function mapTypeLabel(type) {

  if (type === 'BS') return 'Benar / Salah';
  if (type === 'Isian') return 'Isian Singkat';
  if (type === 'Esai') return 'Esai';
  return type;
}

async function saveStudentAnswersToBackend() {
  if (!state.activeAttemptId) return;
  try {
    const data = await apiCall('/student/exam/save-answers', 'POST', {
      attemptId: state.activeAttemptId,
      answers: state.studentAnswers,
      remainingTimeSeconds: state.remainingTimeSeconds
    });
    if (data && data.isLocked === false) {
      document.getElementById('cheat-overlay').classList.add('hidden');
    }
  } catch (err) {}
}

function setupAutosaveTimer() {
  // Save every 60 seconds
  setInterval(saveStudentAnswersToBackend, 60000);
}

function startExamTimer() {
  function updateClock() {
    if (state.remainingTimeSeconds <= -60) {
      clearInterval(state.examTimerInterval);
      showToast('Waktu ujian telah berakhir! Lembar jawaban Anda otomatis dikirim.', 'warning');
      submitExamFinal();
      return;
    }

    state.remainingTimeSeconds--;

    const hrs = Math.floor(state.remainingTimeSeconds / 3600);
    const mins = Math.floor((state.remainingTimeSeconds % 3600) / 60);
    const secs = Math.floor(state.remainingTimeSeconds % 60);

    const pad = (n) => String(n).padStart(2, '0');
    document.getElementById('exam-timer').textContent = `${pad(hrs)}:${pad(mins)}:${pad(secs)}`;
  }

  updateClock();
  if (state.examTimerInterval) clearInterval(state.examTimerInterval);
  state.examTimerInterval = setInterval(updateClock, 1000);
}

function openModalConfirmSubmit() {
  // Check if any unanswered questions or doubt questions exist
  const warningContainer = document.getElementById('submit-summary-warning');
  warningContainer.innerHTML = '';

  let unanswered = 0;
  let doubt = 0;

  state.questions.forEach(q => {
    const ans = state.studentAnswers[q.id];
    const empty = !ans || (Array.isArray(ans) ? !ans.some(val => val !== undefined && val !== null && val !== '') : ans.trim() === '');
    if (empty) unanswered++;
    if (state.studentDoubtQuestions[q.id]) doubt++;
  });

  if (unanswered > 0 || doubt > 0) {
    warningContainer.classList.remove('hidden');
    warningContainer.innerHTML = `
      <i class="fa-solid fa-triangle-exclamation"></i>
      <div>
        <strong>Perhatian Siswa:</strong>
        <ul>
          ${unanswered > 0 ? `<li>Ada <strong>${unanswered}</strong> soal belum diisi jawaban.</li>` : ''}
          ${doubt > 0 ? `<li>Ada <strong>${doubt}</strong> soal ditandai Ragu-ragu (Kuning).</li>` : ''}
        </ul>
      </div>
    `;
  } else {
    warningContainer.classList.add('hidden');
  }

  openModal('modal-confirm-submit');
}

async function extractTextFromPDF(file) {
  const arrayBuffer = await file.arrayBuffer();
  // PDFJS is loaded globally from the CDN tag
  const pdfjsLib = window['pdfjs-dist/build/pdf'];
  pdfjsLib.GlobalWorkerOptions.workerSrc = 'https://cdnjs.cloudflare.com/ajax/libs/pdf.js/3.4.120/pdf.worker.min.js';
  
  const pdf = await pdfjsLib.getDocument({ data: arrayBuffer }).promise;
  let fullText = '';
  for (let i = 1; i <= pdf.numPages; i++) {
    const page = await pdf.getPage(i);
    const textContent = await page.getTextContent();
    const pageText = textContent.items.map(item => item.str).join(' ');
    fullText += `--- Halaman ${i} ---\n` + pageText + '\n';
  }
  return fullText;
}

              <div class="detail-row">
                <span class="detail-label">Kelas</span>
                <span class="detail-colon">:</span>
                <span class="detail-value">${student.classId}</span>
              </div>
              <div class="exam-card-barcode">
                <div class="student-qrcode" id="qrcode-${student.nisn || student.nis}"></div>
              </div>
            </div>
          </div>
        </div>
      </div>
    `;
  });
  htmlContent += '</div>';

  printArea.innerHTML = htmlContent;

  // Generate QR Codes using local qrcode.min.js
  if (window.QRCode) {
    printArea.querySelectorAll('.student-qrcode').forEach(el => {
      const nisnVal = el.id.replace('qrcode-', '');
      new QRCode(el, {
        text: nisnVal,
        width: 50,
        height: 50,
        colorDark: '#1e293b',
        colorLight: '#ffffff',
        correctLevel: QRCode.CorrectLevel.H
      });
    });
  }

  // Close the configuration modal
  closeModal('modal-exam-cards');

  // Wait a brief moment for QR images to generate and then check all image loads
  setTimeout(() => {
    const imgs = printArea.querySelectorAll('img');
    let loadedCount = 0;
    
    const triggerPrint = () => {
      window.print();
      // Clean up
      printArea.innerHTML = '';
    };

    if (imgs.length === 0) {
      setTimeout(triggerPrint, 300);
    } else {
      imgs.forEach(img => {
        if (img.complete) {
          loadedCount++;
          if (loadedCount === imgs.length) {
            setTimeout(triggerPrint, 400);
          }
        } else {
          img.addEventListener('load', () => {
            loadedCount++;
            if (loadedCount === imgs.length) {
              setTimeout(triggerPrint, 400);
            }
          });
          img.addEventListener('error', () => {
            loadedCount++;
            if (loadedCount === imgs.length) {
              setTimeout(triggerPrint, 400);
            }
          });
        }
      });

      // Fallback timer (max 3 seconds) in case of slow or hanging network requests
      setTimeout(() => {
        if (loadedCount < imgs.length) {
          console.warn('Printing fallback triggered: not all images loaded in time.');
          triggerPrint();
        }
      }, 3000);
    }
  }, 150);
}

window.switchAttendanceTab = (tabName) => {
  const scanTab = document.getElementById('att-tab-scan');
  const listTab = document.getElementById('att-tab-list');
  const scanBtn = document.getElementById('tab-att-scan-btn');
  const listBtn = document.getElementById('tab-att-list-btn');
  
  if (!scanTab || !listTab) return;
  
  if (tabName === 'scan') {
    scanTab.classList.remove('hidden');
    listTab.classList.add('hidden');
    scanBtn.className = 'btn btn-primary';
    listBtn.className = 'btn btn-outline-primary';
    
    // Automatically start camera if not running and session is not closed
    const isClosed = !document.getElementById('attendance-closed-indicator').classList.contains('hidden');
    if (!html5QrCode && !isClosed && currentExamAttendanceId) {
      startAttendanceCamera();
    }
  } else {
    scanTab.classList.add('hidden');
    listTab.classList.remove('hidden');
    scanBtn.className = 'btn btn-outline-primary';
    listBtn.className = 'btn btn-primary';
    
    // Stop camera when switching to list tab to save resources
    if (html5QrCode) {
      stopAttendanceCamera();
    }
  }
};

let popupTimeout = null;

function toTitleCase(str) {
  if (!str) return '';
  let result = str.toLowerCase().split(' ').map(word => {
    return word.charAt(0).toUpperCase() + word.slice(1);
  }).join(' ');

  // Pengecualian gelar
  const degrees = {
    'S.pd': 'S.Pd', 'M.pd': 'M.Pd',
    'S.kom': 'S.Kom', 'M.kom': 'M.Kom',
    'S.ag': 'S.Ag', 'M.ag': 'M.Ag',
    'S.si': 'S.Si', 'M.si': 'M.Si',
    'S.t': 'S.T', 'M.t': 'M.T',
    'S.e': 'S.E', 'M.e': 'M.E',
    'S.ab': 'S.Ab', 'S.sos': 'S.Sos',
    'M.sos': 'M.Sos', 'S.h': 'S.H', 'M.h': 'M.H'
  };

  for (let key in degrees) {
    const regex = new RegExp(key.replace(/\./g, '\\.'), 'g');
    result = result.replace(regex, degrees[key]);
  }

  return result;
}

function showAttendancePopup(name) {
  const popup = document.getElementById('attendance-scan-popup');
  const nameSpan = document.getElementById('attendance-popup-student-name');
  if (!popup || !nameSpan) return;
  
  nameSpan.textContent = toTitleCase(name);
  popup.classList.remove('hidden');
  
  // Force reflow
  void popup.offsetWidth;
  
  popup.style.opacity = '1';
  popup.style.transform = 'translateY(0)';
  
  if (popupTimeout) {
    clearTimeout(popupTimeout);
  }
  
  popupTimeout = setTimeout(() => {
    popup.style.opacity = '0';
    popup.style.transform = 'translateY(-20px)';
    
    popupTimeout = setTimeout(() => {
      popup.classList.add('hidden');
    }, 300);
  }, 3000);
}

// Start camera scanner using html5-qrcode
async function startAttendanceCamera() {
  if (!window.isSecureContext) {
    showToast('Kamera diblokir karena koneksi tidak aman (Non-HTTPS/HTTP).', 'danger', 7000);
    alert('Akses Kamera Gagal (Koneksi Tidak Aman)\n\nBrowser HP memblokir kamera jika website diakses lewat HTTP (misal IP komputer Anda: http://192.168.x.x:3000).\n\nCara mengatasi:\n1. Gunakan komputer/laptop (localhost diperbolehkan oleh browser tanpa HTTPS).\n2. Aktifkan HTTPS (misal lewat Ngrok, Cloudflare Tunnel, atau lokal HTTPS).\n3. Khusus HP Android & Google Chrome:\n   - Buka: chrome://flags/#unsafely-treat-insecure-origin-as-secure\n   - Masukkan alamat IP komputer Anda (contoh: http://192.168.1.10:3000)\n   - Pilih "Enabled" lalu klik "Relaunch".');
    return;
  }

  const attendanceReader = document.getElementById('attendance-reader-wrapper');
  if (attendanceReader) {
    attendanceReader.style.display = 'block';
  }
  
  const startBtn = document.getElementById('btn-start-camera');
  if (startBtn) startBtn.classList.add('hidden');
  const stopBtn = document.getElementById('btn-stop-camera');
  if (stopBtn) stopBtn.classList.add('hidden'); // Always hide stop camera button

  html5QrCode = new Html5Qrcode("attendance-reader");

  try {
    const cameras = await Html5Qrcode.getCameras();
    const cameraSelect = document.getElementById('attendance-camera-select');
    
    if (cameras && cameras.length > 0) {
      if (cameraSelect) {
        cameraSelect.innerHTML = '';
        cameras.forEach((camera) => {
          const option = document.createElement('option');
          option.value = camera.id;
          option.textContent = camera.label || `Kamera ${cameraSelect.options.length + 1}`;
          cameraSelect.appendChild(option);
        });
        cameraSelect.classList.add('hidden'); // Always keep camera choice dropdown hidden
      }
      
      // Try to find the back camera
      let selectedCameraId = cameras[0].id;
      const backCam = cameras.find(c => {
        const label = (c.label || '').toLowerCase();
        return label.includes('back') || label.includes('rear') || label.includes('environment') || label.includes('belakang') || label.includes('samping');
      });
      
      if (backCam) {
        selectedCameraId = backCam.id;
      } else if (cameras.length > 1) {
        selectedCameraId = cameras[cameras.length - 1].id;
      }
      
      cameraSelect.value = selectedCameraId;

      cameraSelect.onchange = async () => {
        if (html5QrCode) {
          try {
            await html5QrCode.stop();
          } catch(e) {}
          
          await html5QrCode.start(
            cameraSelect.value,
            {
              fps: 10,
              qrbox: function(viewfinderWidth, viewfinderHeight) {
                const minEdge = Math.min(viewfinderWidth, viewfinderHeight);
                const size = Math.floor(minEdge * 0.8);
                const root = document.documentElement;
                root.style.setProperty('--qrbox-size', size + 'px');
                root.style.setProperty('--qrbox-top', ((viewfinderHeight - size) / 2) + 'px');
                root.style.setProperty('--qrbox-left', ((viewfinderWidth - size) / 2) + 'px');
                return { width: size, height: size };
              }
            },
            async (decodedText) => {
              await submitAttendanceScan(decodedText);
            },
            (errorMessage) => {}
          ).catch(err => {
            showToast('Gagal mengganti kamera: ' + err, 'danger');
          });
        }
      };

      await html5QrCode.start(
        selectedCameraId,
        {
          fps: 10,
          qrbox: function(viewfinderWidth, viewfinderHeight) {
            const minEdge = Math.min(viewfinderWidth, viewfinderHeight);
            const size = Math.floor(minEdge * 0.8);
            const root = document.documentElement;
            root.style.setProperty('--qrbox-size', size + 'px');
            root.style.setProperty('--qrbox-top', ((viewfinderHeight - size) / 2) + 'px');
            root.style.setProperty('--qrbox-left', ((viewfinderWidth - size) / 2) + 'px');
            return { width: size, height: size };
          }
        },
        async (decodedText) => {
          await submitAttendanceScan(decodedText);
        },
        (errorMessage) => {}
      );
    } else {
      await startAttendanceCameraWithConstraint();
    }
  } catch (err) {
    console.warn("getCameras failed, falling back to facingMode constraint", err);
    await startAttendanceCameraWithConstraint();
  }
}

async function startAttendanceCameraWithConstraint() {
  try {
    await html5QrCode.start(
      { facingMode: "environment" },
      {
        fps: 10,
        qrbox: { width: 250, height: 250 }
      },
      async (decodedText) => {
        await submitAttendanceScan(decodedText);
      },
      (errorMessage) => {}
    );
  } catch (err) {
    showToast('Gagal mengakses kamera: ' + err, 'danger');
    stopAttendanceCamera();
  }
}

async function stopAttendanceCamera() {
  const startBtn = document.getElementById('btn-start-camera');
  if (startBtn) startBtn.classList.add('hidden');
  const stopBtn = document.getElementById('btn-stop-camera');
  if (stopBtn) stopBtn.classList.add('hidden');
  
  const reader = document.getElementById('attendance-reader-wrapper');
  if (reader) reader.style.display = 'none';
  
  const cameraSelect = document.getElementById('attendance-camera-select');
  if (cameraSelect) {
    cameraSelect.classList.add('hidden');
    cameraSelect.innerHTML = '';
    cameraSelect.onchange = null;
  }
  
  if (html5QrCode) {
    try {
      await html5QrCode.stop();
    } catch (e) {}
    html5QrCode = null;
  }
}

async function loadAttendanceData() {
  if (!currentExamAttendanceId) return;
  
  try {
    const data = await apiCall(`/teacher/exams/${currentExamAttendanceId}/attendance`);
    attendanceData = data.attendance;
    currentExamAttendanceClosed = data.exam.attendanceClosed;
    
    // Set exam name
    document.getElementById('attendance-exam-name').textContent = `${data.exam.title} (${data.exam.classId})`;
    
    // Check if session is closed
    if (currentExamAttendanceClosed) {
      document.getElementById('attendance-closed-indicator').classList.remove('hidden');
      document.getElementById('btn-close-attendance-session').classList.add('hidden');
      await stopAttendanceCamera();
      document.getElementById('btn-start-camera').classList.add('hidden');
    }
    
    renderAttendanceTable();
  } catch (err) {
    showToast('Gagal memuat data kehadiran.', 'danger');
  }
}

function renderAttendanceTable() {
  const tbody = document.getElementById('table-attendance-body');
  tbody.innerHTML = '';
  
  const searchVal = document.getElementById('attendance-search-input').value.trim().toLowerCase();
  
  let filtered = attendanceData;
  if (searchVal) {
    filtered = attendanceData.filter(a => 
      a.name.toLowerCase().includes(searchVal) || 
      a.nis.toLowerCase().includes(searchVal) ||
      (a.nisn && a.nisn.toLowerCase().includes(searchVal))
    );
  }
  
  // Count stats
  let total = attendanceData.length;
  let present = attendanceData.filter(a => a.status === 'Hadir').length;
  let absentYet = attendanceData.filter(a => a.status === 'Belum Hadir').length;
  let absentLocked = attendanceData.filter(a => ['Tidak Hadir', 'Sakit', 'Izin'].includes(a.status)).length;
  
  document.getElementById('att-stat-total').textContent = total;
  document.getElementById('att-stat-present').textContent = present;
  document.getElementById('att-stat-absent-yet').textContent = absentYet;
  document.getElementById('att-stat-absent-locked').textContent = absentLocked;

  if (filtered.length === 0) {
    tbody.innerHTML = `<tr><td colspan="6" class="text-center text-muted" style="padding:20px;">Tidak ada siswa ditemukan</td></tr>`;
    return;
  }
  
  filtered.forEach((a, idx) => {
    let statusBadge = '';
    const isLocked = currentExamAttendanceClosed;
    
    if (!isLocked) {
      statusBadge = `
        <select class="form-control" style="width: auto; display: inline-block; padding: 2px 8px; font-size: 0.85rem;" onchange="handleManualAttendance('${a.nisn || a.nis}', this.value)">
          <option value="Belum Hadir" ${a.status === 'Belum Hadir' ? 'selected' : ''}>Belum Hadir</option>
          <option value="Hadir" ${a.status === 'Hadir' ? 'selected' : ''}>Hadir</option>
          <option value="Sakit" ${a.status === 'Sakit' ? 'selected' : ''}>Sakit</option>
          <option value="Izin" ${a.status === 'Izin' ? 'selected' : ''}>Izin</option>
          <option value="Tidak Hadir" ${a.status === 'Tidak Hadir' ? 'selected' : ''}>Tidak Hadir</option>
        </select>
      `;
    } else {
      if (a.status === 'Hadir') {
        statusBadge = `<span class="badge badge-success" style="display: inline-flex; align-items: center; gap: 4px;">
          <svg style="width: 12px; height: 12px; fill: none; stroke: currentColor; stroke-width: 3;" viewBox="0 0 24 24">
            <path stroke-linecap="round" stroke-linejoin="round" d="M5 13l4 4L19 7" />
          </svg> Hadir
        </span>`;
      } else if (a.status === 'Tidak Hadir') {
        statusBadge = `<span class="badge badge-danger" style="display: inline-flex; align-items: center; gap: 4px;">
          <svg style="width: 12px; height: 12px; fill: none; stroke: currentColor; stroke-width: 3;" viewBox="0 0 24 24">
            <path stroke-linecap="round" stroke-linejoin="round" d="M6 18L18 6M6 6l12 12" />
          </svg> Tidak Hadir
        </span>`;
      } else if (a.status === 'Sakit') {
        statusBadge = `<span class="badge" style="background-color: #f6e05e; color: #744210; display: inline-flex; align-items: center; gap: 4px;">
          <i class="fa-solid fa-bed-pulse"></i> Sakit
        </span>`;
      } else if (a.status === 'Izin') {
        statusBadge = `<span class="badge" style="background-color: #90cdf4; color: #2a4365; display: inline-flex; align-items: center; gap: 4px;">

      } else {
        statusBadge = `<span class="badge badge-warning" style="background-color: #e2e8f0; color: #4a5568; display: inline-flex; align-items: center; gap: 4px;">
          <svg style="width: 12px; height: 12px; fill: none; stroke: currentColor; stroke-width: 3;" viewBox="0 0 24 24">

      }
    }
    
    const scanTime = a.scannedAt ? new Date(a.scannedAt).toLocaleTimeString('id-ID', { hour: '2-digit', minute: '2-digit', second: '2-digit' }) : '-';
    
    tbody.innerHTML += `

// Speak name using browser Text-to-Speech (SpeechSynthesis API)
function speakAttendance(name, alreadyScanned = false) {
  if ('speechSynthesis' in window) {
    window.speechSynthesis.cancel();
    
    // Convert name to Title Case to prevent spelling letter-by-letter (uppercase words are spelled out by some TTS)
    const titleCaseName = toTitleCase(name);
    
    const text = alreadyScanned ? `${titleCaseName} Sudah Hadir` : `${titleCaseName} Hadir`;
    const utterance = new SpeechSynthesisUtterance(text);
    utterance.lang = 'id-ID';
    utterance.rate = 0.95;
    utterance.pitch = 1.0;
    
    // Find Indonesian voice if available
    const voices = window.speechSynthesis.getVoices();
    const idVoice = voices.find(v => v.lang.startsWith('id') || v.lang.toLowerCase().includes('indonesia'));
    if (idVoice) {
      utterance.voice = idVoice;
    }
    
    window.speechSynthesis.speak(utterance);
  }
}

// Submit manual/QR code scan
async function submitAttendanceScan(nisnCode) {
  const nisn = String(nisnCode || '').trim();
  if (!nisn) return;
  
  const now = Date.now();
  if (nisn === lastScannedCode && (now - lastScannedTime) < 5000) {
    console.log(`Scan duplicate blocked by 5s cooldown for NISN: ${nisn}`);
    return;
  }
  
  lastScannedCode = nisn;
  lastScannedTime = now;
  
  try {
    const res = await fetch(`${API_BASE}/teacher/exams/${currentExamAttendanceId}/attendance/scan`, {
      method: 'POST',
      headers: {
        'Authorization': state.token || localStorage.getItem('cbt_token'),
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({ nisn })
    });
    
    if (!res.ok) {
      const err = await res.json().catch(() => ({ message: 'NISN atau siswa tidak valid.' }));
      showToast(err.message, 'danger');
      return;
    }
    
    const result = await res.json();
    
    // Play voice over
    speakAttendance(result.student.name, result.alreadyScanned);
    
    // Show popup overlay
    showAttendancePopup(result.student.name);
    
    // Reload lists
    await loadAttendanceData();
  } catch (err) {
    showToast('Gagal memproses absensi.', 'danger');
  }
}

// Close session and lock unscanned as Tidak Hadir
async function closeAttendanceSession() {
  if (!confirm('Tutup sesi absensi ujian ini?\n\nSemua siswa yang BELUM HADIR akan dikunci secara permanen dengan status "Tidak Hadir" di database, dan pengeditan absensi akan ditutup.\n\nTindakan ini tidak dapat dibatalkan.')) {
    return;
  }
  
  try {
    const res = await apiCall(`/teacher/exams/${currentExamAttendanceId}/attendance/close`, 'POST');
    showToast(res.message);
    await stopAttendanceCamera();
    await loadAttendanceData();
  } catch (err) {
    showToast('Gagal menutup sesi absensi.', 'danger');
  }
}

window.handleManualAttendance = async (nisn, newStatus) => {
  try {
    const res = await apiCall(`/teacher/exams/${currentExamAttendanceId}/attendance/manual`, 'POST', {
      nisn,
      status: newStatus
    });
    if (res.success) {
      showToast(`Kehadiran ${res.student.name} diperbarui menjadi ${res.student.status}.`, 'success');
      loadAttendanceData();
    }
  } catch (err) {
    showToast(err.message || 'Gagal memperbarui status manual.', 'danger');
    loadAttendanceData(); // reload to revert select
  }
};
